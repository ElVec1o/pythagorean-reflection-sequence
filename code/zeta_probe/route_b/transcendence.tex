\section{Transcendence of the relaxed growth series}
\label{sec:transcendence}

Throughout, $q=x^{2}$, and we study the \emph{relaxed} ordinary generating
function $V(x)=\sum_{n\ge0}v_n x^{n}$ of \cite[A396406]{OEIS} through the
block decomposition of Route~B (\S\ref{sec:routeb}). Recall the two scalar
series in $q$ that carry the singular behaviour of the relaxed model.

\paragraph{Bulk block.} With
\begin{equation}\label{eq:alphagamma}
  \alpha_k(q)=\frac{2q^{\,k+1}}{1-q^{\,k+1}},\qquad
  \gamma_k(q)=\frac{2q^{\,k+2}}{1-q^{\,k+2}}-\frac{2q^{\,k+1}}{1-q^{\,k+1}}
            =\alpha_{k+1}(q)-\alpha_k(q),
\end{equation}
set
\begin{equation}\label{eq:Sk}
  S_k(q)=\sum_{j\ge0}\alpha_{k+2j}(q)\prod_{i=0}^{j-1}\gamma_{k+2i}(q),
  \qquad
  G_0(q)=\frac{S_0(q)}{1-S_1(q)} .
\end{equation}
$G_0$ is the bulk-run block; its Taylor coefficients are
$2,2,6,2,18,6,42,18,118,50,282,190,706,594,\dots$ (verified).

\paragraph{Travel block.} With
\begin{equation}\label{eq:AC}
  A_k(q)=\frac{2q}{1-q^{\,k+1}},\qquad
  C_k(q)=\frac{2q^{\,k+3}}{1-q^{\,k+2}}-\frac{2q^{\,k+2}}{1-q^{\,k+1}},
\end{equation}
set $\Sigma_k(q)=\sum_{j\ge0}A_{k+2j}(q)\prod_{i=0}^{j-1}C_{k+2i}(q)$ and
$G^{T}_0(q)=\Sigma_0(q)/(1-\Sigma_1(q))$. The dominant singularity of the
relaxed model is the smallest positive root of $\Sigma_1(q)=1$, at
$q^{\ast}=0.449453630558948\ldots$, giving the growth rate
$1/\sqrt{q^{\ast}}=1.4916177871\ldots$.

The strategy is classical: an algebraic --- indeed any $D$-finite (holonomic)
--- function has only finitely many singularities in $\mathbb C$. We prove that
$G_0$ and $G^{T}_0$ each have infinitely many poles in the unit disc
$|q|<1$, accumulating at $q=1$, and deduce transcendence over $\mathbb C(q)$,
hence over $\mathbb C(x)$.

\subsection{Analyticity and the pole criterion}

\begin{lemma}[Holomorphy on the disc]\label{lem:analytic}
$S_0,S_1,\Sigma_0,\Sigma_1$ are holomorphic on $\{|q|<1\}$; in particular they
are real-analytic on $(0,1)$. Each is meromorphic on $\mathbb C$ with poles only
on $|q|=1$ \textup(at roots of unity\textup).
\end{lemma}

\begin{proof}
Fix $r\in(0,1)$. For $|q|\le r$ and $m\ge1$ we have $|1-q^{m}|\ge 1-r^{m}\ge
1-r>0$, so every $\alpha_k$ in \eqref{eq:alphagamma} is holomorphic on the closed
disc $\overline{D_r}$ with $|\alpha_k(q)|\le 2r^{\,k+1}/(1-r)$ and hence
$|\gamma_k(q)|\le 4r^{\,k+1}/(1-r)$; the same bounds hold for $A_k,C_k$ in
\eqref{eq:AC}. Consequently the $j$-th summand of \eqref{eq:Sk} is bounded on
$\overline{D_r}$ by
\[
  |\alpha_{k+2j}|\prod_{i<j}|\gamma_{k+2i}|
  \;\le\;\frac{2r^{\,k+2j+1}}{1-r}\prod_{i=0}^{j-1}\frac{4r^{\,k+2i+1}}{1-r}
  \;=\;O\!\Bigl(r^{\,2j^{2}}\Bigr),
\]
which is summable and super-geometrically small. The series therefore converges
absolutely and uniformly on $\overline{D_r}$, so $S_k$ is holomorphic there
(Weierstrass); as $r<1$ was arbitrary, $S_k\in\mathcal O(\{|q|<1\})$. Each
summand is a rational function whose poles are roots of $1-q^{m}=0$, i.e.\ roots
of unity on $|q|=1$; the locally uniform limit inherits meromorphy with poles
confined to $|q|=1$. The argument for $\Sigma_k$ is identical.
\end{proof}

\begin{corollary}[Pole criterion]\label{cor:polecrit}
If $q_0\in(0,1)$ satisfies $S_1(q_0)=1$ and $S_0(q_0)\ne0$, then $q_0$ is a pole
of $G_0$. The analogous statement holds for $G^{T}_0$ with $\Sigma_1,\Sigma_0$.
\end{corollary}

\begin{proof}
By Lemma~\ref{lem:analytic}, $S_0$ and $1-S_1$ are holomorphic near $q_0$; the
hypothesis makes $q_0$ a zero of the denominator and not of the numerator, so
$G_0=S_0/(1-S_1)$ has a pole at $q_0$ (of order equal to the vanishing order of
$1-S_1$). \end{proof}

Thus it suffices to produce infinitely many zeros of $1-S_1$ (resp.\ $1-\Sigma_1$)
in $(0,1)$ at which the numerator does not vanish.

\subsection{The oscillation near \texorpdfstring{$q=1$}{q=1}}

Put $q=e^{-\tau}$, $\tau=-\ln q>0$, so $\alpha_k(q)=2/(e^{(k+1)\tau}-1)$ and
$\gamma_k(q)=\alpha_{k+1}-\alpha_k<0$. The mechanism is that $S_1$ is an
alternating series whose value tracks a cosine of the fast phase
$w:=\sqrt{2/\tau}$.

\begin{lemma}[Sign structure]\label{lem:signs}
For $q\in(0,1)$ one has $\alpha_k(q)>0$, $\gamma_k(q)<0$, and the $j$-th term of
\eqref{eq:Sk} equals $(-1)^{j}t_j(q)$ with
$t_j(q)=\alpha_{1+2j}\prod_{i<j}\lvert\gamma_{1+2i}\rvert>0$. Hence
$S_1(q)=\sum_{j\ge0}(-1)^{j}t_j(q)$ is an alternating series of positive terms.
The same holds for $\Sigma_1$ \textup(with $C_k<0$ on $(0,1)$\textup).
\end{lemma}

\begin{proof}
Immediate from \eqref{eq:alphagamma}: $0<q^{k+1}<1$ gives $\alpha_k>0$, and
$0<1-q^{k+2}<1-q^{k+1}$ for $0<q<1$ gives
$\tfrac{2q^{k+2}}{1-q^{k+2}}<\tfrac{2q^{k+1}}{1-q^{k+1}}$, i.e.\ $\gamma_k<0$;
thus $\prod_{i<j}\gamma_{1+2i}=(-1)^{j}\prod_{i<j}|\gamma_{1+2i}|$. For
\eqref{eq:AC}, $C_k=\tfrac{2q^{k+3}}{1-q^{k+2}}-\tfrac{2q^{k+2}}{1-q^{k+1}}
<\tfrac{2q^{k+2}}{1-q^{k+2}}-\tfrac{2q^{k+2}}{1-q^{k+1}}<0$ likewise.
\end{proof}

The exact telescoping
$\prod_{i=0}^{j-1}\bigl[\tfrac1{2i+2}-\tfrac1{2i+3}\bigr]
   =\prod_{i=0}^{j-1}\tfrac1{(2i+2)(2i+3)}=\tfrac1{(2j+1)!}$
identifies the leading model of $t_j$. Writing $g(y)=y/(e^{y}-1)$ (so
$\alpha_k=\tfrac{2}{(k+1)\tau}\,g((k+1)\tau)$ and
$1-\tfrac y2\le g(y)\le e^{-y/2}\le1$ for $y>0$), one finds that in the regime
$k\tau\to0$ governing the dominant terms ($k\sim w=\sqrt{2/\tau}$),
\[
  t_j(q)=\frac{(2/\tau)^{\,j+1}}{(2j+2)!}\,\bigl(1+o(1)\bigr),
  \qquad
  \sum_{j\ge0}(-1)^{j}\frac{(2/\tau)^{\,j+1}}{(2j+2)!}
  =\sum_{m\ge1}(-1)^{m-1}\frac{w^{2m}}{(2m)!}=1-\cos w .
\]
This heuristic is turned into an \emph{exact} reduction by the following
domination and summation-by-parts identity, after which the only missing input is
a single oscillatory-average bound (Remark~\ref{rem:gap}).

\begin{lemma}[Monotone domination]\label{lem:dom}
Write $\hat t_j(q)=\dfrac{(2/\tau)^{\,j+1}}{(2j+2)!}$ for the leading model term, so
that, exactly,
\[
  \sum_{j\ge0}(-1)^{j}\hat t_j=1-\cos w,\qquad w=\sqrt{2/\tau}.
\]
Then for all $q\in(0,1)$ and all $j$ one has $0<t_j\le\hat t_j$; equivalently
$\rho_j:=t_j/\hat t_j\in(0,1]$. Moreover $\rho_j$ is strictly decreasing in $j$
with $\rho_j\to0$.
\end{lemma}

\begin{proof}
With $g(y)=y/(e^{y}-1)$ one has $\alpha_k=\tilde\alpha_k\,g((k{+}1)\tau)$ where
$\tilde\alpha_k=2/((k{+}1)\tau)$, and $0<g\le1$ gives $\alpha_{1+2j}\le\tilde
\alpha_{1+2j}$. For the difference factor, the discrete derivative of
$s\mapsto 2/(e^{s\tau}-1)$ has $-\,\partial_s=\dfrac{2\tau e^{s\tau}}{(e^{s\tau}-1)^2}
=\tfrac{2}{\tau}\,h(s\tau)^2/s^2$ with $h(y)=\tfrac{y/2}{\sinh(y/2)}\in(0,1]$, while the
model derivative is $\tfrac{2}{\tau}/s^{2}$; integrating over a unit step gives
$0<|\gamma_{1+2i}|=\alpha_{1+2i}-\alpha_{2+2i}\le\tilde\alpha_{1+2i}-\tilde
\alpha_{2+2i}$. Multiplying these termwise bounds yields $0<t_j\le\hat t_j$.
Finally $\rho_j/\rho_{j-1}=g((2j{+}2)\tau)\,|\gamma_{1+2(j-1)}|/(\tilde\alpha_{2j-1}
-\tilde\alpha_{2j})<1$, and $\rho_j$ is a product of $j$ factors each $\le h(\cdot)^2<1$
times $g\to0$, so $\rho_j\downarrow0$.
\end{proof}

\begin{lemma}[Exact cosine reduction]\label{lem:reduce}
Let $\mu_j:=\rho_{j-1}-\rho_j\ge0$ \textup($\rho_{-1}:=1$\textup), a probability
weight \textup($\mu_j\ge0$, $\sum_j\mu_j=1$\textup), and let
$R_j(w)=\cos w-\sum_{i=0}^{j}\tfrac{(-1)^{i}w^{2i}}{(2i)!}$ be the order-$2j$
Taylor remainder of the cosine. Then, for every $q\in(0,1)$,
\begin{equation}\label{eq:reduce}
  S_1(q)=\bigl(1-\cos w\bigr)+\sum_{j\ge0}\mu_j\,R_j(w),
\end{equation}
and identically for $\Sigma_1$. In particular $S_1$ equals the $\mu$-average
$\sum_j\mu_j A_{j-1}$ of the partial sums $A_{j-1}=\sum_{i<j}(-1)^{i}\hat t_i$ of the
model series.
\end{lemma}

\begin{proof}
Abel summation of $S_1=\sum_j(-1)^{j}t_j=\sum_j(-1)^{j}\hat t_j\rho_j$ with
$\rho_j\downarrow0$ (Lemma~\ref{lem:dom}) gives $S_1=\sum_j A_j(\rho_j-\rho_{j+1})
=\sum_j\mu_{j+1}A_j=\sum_j\mu_jA_{j-1}$, the boundary term $A_m\rho_{m+1}\to0$ because
$A_m\to1-\cos w$ is bounded while $\rho_{m+1}\to0$. Since
$A_{j-1}=1-\sum_{i=0}^{j}\tfrac{(-1)^iw^{2i}}{(2i)!}=(1-\cos w)+R_j(w)$ and
$\sum_j\mu_j=1$, \eqref{eq:reduce} follows. \end{proof}

\noindent Identity~\eqref{eq:reduce} is verified to $110$ decimal digits in
\texttt{transcendence\_verify\_ALL.py}. It reduces Lemma~\ref{lem:cos} to the single
estimate $\sum_j\mu_jR_j(w)=O(\sqrt\tau)$ uniformly in the phase. The leading term is
moreover obtained in \emph{closed form}. Writing $S_1-(1-\cos w)=-\sum_j(-1)^j\hat
t_j\,\delta_j$ with $\delta_j=1-\rho_j$, the uniform Taylor expansion of $\log\rho_j$
in the saddle window $j=O(\tau^{-1/2})$ is
\[
  \log\rho_j=-(j{+}1)\tau-\tfrac{1}{9}\tau^{2}j^{3}+O(\tau^{2}j^{2}+\tau^{3}j^{5}),
\]
the linear term from the single end-factor $g(y)=y/(e^{y}-1)$ and the cubic from the
cumulative product of $h(y)^2=(y/2)^2/\sinh^2(y/2)$ (with the exact difference-ratio
correction $-y\tau/12$). With the operator $\vartheta=\tfrac12 w\partial_w$, the exact
resummations $\sum_{n\ge1}(-1)^{n-1}n^{m}\tfrac{w^{2n}}{(2n)!}=\vartheta^{m}(1-\cos w)$
give $\vartheta(1-\cos w)=\tfrac{w}{2}\sin w$ and $\vartheta^{3}(1-\cos
w)=-\tfrac{w^{3}}{8}\sin w+O(w^{2})$, whence the linear and cubic terms contribute
$-\tfrac1{\sqrt2}\sqrt\tau\sin w$ and $+\tfrac{\sqrt2}{36}\sqrt\tau\sin w$ respectively
(using $\tau w=\sqrt{2\tau}$, $\tau^{2}w^{3}=2\sqrt2\,\sqrt\tau$). Hence
\begin{equation}\label{eq:c1}
  S_1(q)=1-\cos w-\frac{17\sqrt2}{36}\,\sqrt\tau\,\sin w+O(\tau),\qquad
  c_1=-\frac{17\sqrt2}{36}=-0.6678230711\ldots,
\end{equation}
the constant confirmed to $15$ digits by Richardson extrapolation of $(S_1-1)/\sqrt\tau$
at the phases $\sin w=1$ (\texttt{transcendence\_verify\_ALL.py}). The derivation is
rigorous term by term; the one step not made uniform is the resummation of the
$O(\tau)$ remainder (a Watson/saddle estimate), which is exactly the content of
Remark~\ref{rem:gap}.

\subsection{The numerator amplitude: \texorpdfstring{$S_0\sim w\sin w$}{S0 ~ w sin w}}
\label{ss:numerator}

Lemma~\ref{lem:cos} controls the \emph{denominator} $1-S_1$: it has infinitely many
real zeros $q_m\uparrow1$, the phases $w_m=\sqrt{2/\tau_m}\to(m-\tfrac12)\pi$. By
Corollary~\ref{cor:polecrit} each $q_m$ is a genuine pole of $G_0=S_0/(1-S_1)$ \emph{provided}
$S_0(q_m)\ne0$. We now show the numerator is governed by the \emph{same} saddle engine,
which makes $S_0(q_m)\ne0$ for all large $m$ a theorem rather than a finite check.

The telescoped numerator is the \emph{even} ladder
$S_0=\sum_{j\ge0}(-1)^{j}s_j$, $s_j=\alpha_{2j}\prod_{i<j}|\gamma_{2i}|$, against the
\emph{odd} ladder $S_1=\sum_{j\ge0}(-1)^{j}t_j$, $t_j=\alpha_{1+2j}\prod_{i<j}|\gamma_{1+2i}|$
of Lemma~\ref{lem:dom}. With $\alpha_k=2/(e^{(k+1)\tau}-1)=\tfrac{2}{(k+1)\tau}g\!\big((k{+}1)\tau\big)$,
$g(y)=y/(e^y-1)\in(0,1]$, and $|\gamma_k|=\alpha_k-\alpha_{k+1}\sim\tfrac{2}{\tau(k+1)(k+2)}$,
the leading models of the two ladders are
\begin{equation}\label{eq:hatS0}
  \hat t_j=\frac{(2/\tau)^{j+1}}{(2j+2)!},\qquad
  \hat s_j=\frac{(2/\tau)^{j+1}}{(2j+1)!}=(2j+2)\,\hat t_j,
\end{equation}
the only change being the parity of the factorial: $\prod_{i<j}(2i{+}1)(2i{+}2)=(2j)!$ for
the even ladder versus $\prod_{i<j}(2i{+}2)(2i{+}3)=(2j+1)!$ for the odd one. Exactly as
$\sum_j(-1)^j\hat t_j=1-\cos w$, one has the companion closed form
\begin{equation}\label{eq:wsinw}
  \sum_{j\ge0}(-1)^{j}\hat s_j
  =\sum_{j\ge0}(-1)^{j}\frac{w^{2j+2}}{(2j+1)!}
  =w\sin w ,
\end{equation}
verified to $20$ digits in \texttt{sigma0\_layer\_derivation.py}; the amplitude factor
$\hat s_j/\hat t_j=2j+2$ is precisely what turns $1-\cos w=\sum(-1)^{m-1}\tfrac{w^{2m}}{(2m)!}$
into $w\sin w=\sum(-1)^{m-1}(2m)\tfrac{w^{2m}}{(2m)!}=2\vartheta(1-\cos w)$.

\begin{lemma}[Numerator reduction]\label{lem:numred}
The form factors $\sigma_j:=s_j/\hat s_j$ satisfy $\sigma_j\in(0,1]$, strictly decreasing
with $\sigma_j\to0$ \textup(verbatim the proof of Lemma~\ref{lem:dom}, the bound coming from
the same $g\le1$ and $h(y)^2=(y/2)^2/\sinh^2(y/2)<1$\textup). Writing
$\nu_j:=\sigma_{j-1}-\sigma_j\ge0$ \textup($\sigma_{-1}:=1$, $\sum_j\nu_j=1$\textup) and
$\widetilde R_j(w):=w\sin w-\sum_{i=0}^{j}(-1)^{i}\tfrac{w^{2i+2}}{(2i+1)!}$ for the
order-$(2j{+}1)$ sine remainder, Abel summation gives, exactly,
\begin{equation}\label{eq:numred}
  S_0(q)=w\sin w+\sum_{j\ge0}\nu_j\,\widetilde R_j(w).
\end{equation}
\end{lemma}

\noindent The correction in \eqref{eq:numred} is the image of the
Lemma~\ref{lem:reduce} error under the amplitude operator $2\vartheta$: with
$\delta_j=1-\sigma_j$ and \eqref{eq:hatS0},
$S_0-w\sin w=-\sum_j(-1)^j\hat s_j\delta_j=-\sum_j(-1)^j(2j+2)\hat t_j\delta_j
=2\vartheta\big(S_1-(1-\cos w)\big)+o(1)$. Applying $2\vartheta$ to the closed-form
leading term \eqref{eq:c1} --- using $2\vartheta(\sqrt\tau\sin w)=\sqrt2\cos w+o(1)$
(since $\sqrt\tau\,\sin w=\sqrt2\,w^{-1}\sin w$ and $\vartheta(w^{-1}\sin w)
=\tfrac12(\cos w-w^{-1}\sin w)$) --- yields the closed-form companion of \eqref{eq:c1}:
\begin{equation}\label{eq:numc1}
  S_0(q)=w\sin w-\frac{17}{18}\cos w+o(1),
  \qquad \tfrac{17}{18}=\tfrac{17\sqrt2}{36}\cdot\sqrt2 ,
\end{equation}
the \emph{same} constant $17$ propagating from \eqref{eq:c1}; \eqref{eq:numc1} is confirmed
to $3$--$4$ digits across $\tau\in[1.5\times10^{-4},2\times10^{-2}]$ in
\texttt{sigma0\_layer\_derivation.py}, with the residual decaying. \big(The travel
block $\Sigma_0$ obeys the identical leading law $\Sigma_0=w\sin w+O(1)$, with its own
subleading constant $+\tfrac1{18}\cos w$ from the travel value $c_1=+\tfrac{\sqrt2}{36}$
of Remark~\ref{rem:blockscope}.\big)

The sharp subleading $-\tfrac{17}{18}\cos w$ in \eqref{eq:numc1} is the saddle value; but the
\emph{non-vanishing} needs only the much weaker $S_0-w\sin w=o(w)$, and \emph{that is unconditional},
by the absolute contour of Lemma~\ref{lem:T2abs} --- \emph{no} Watson/saddle resummation. Indeed the
numerator ladder differs from the denominator only by the amplitude factor $\hat s_j/\hat t_j=2j+2$
\eqref{eq:hatS0}; inserting it into the representation
$T_2=\sum_{n\ge1}(-1)^nh(n)=\tfrac1{2\pi i}\oint_{\partial R}h\,\tfrac{\pi}{\sin\pi s}\,ds$ of
Lemma~\ref{lem:T2abs} weights that contour integral by $2s$,
\[
   \sum_{n\ge1}(-1)^n(2n)\,h(n)=\frac1{2\pi i}\oint_{\partial R}2s\,h(s)\,\frac{\pi}{\sin\pi s}\,ds
   \qquad\bigl(\operatorname{Res}_{s=n}2s\,h\,\tfrac{\pi}{\sin\pi s}=(-1)^n2n\,h(n)\bigr),
\]
while the elementary remainder $2\vartheta T_1=W\sin W-w\sin w$ has modulus
$\le|W-w|+w|\sin W-\sin w|\le\sqrt{\tfrac\tau2}+w^2(1-e^{-\tau/2})=O(1)$. Bounding the integral in modulus
exactly as in Lemma~\ref{lem:T2abs}, the extra $2|s|\le W(1+o(1))\sim\sqrt2\,\tau^{-1/2}$ on the
left-edge mass turns its $O(\tau^{1/4})$ into
\[
   \Bigl|\sum_{n\ge1}(-1)^n2n\,h(n)\Bigr|\ \le\ \frac1{2\pi}\oint_{\partial R}2|s|\,\Bigl|h\,\tfrac{\pi}{\sin\pi s}\Bigr|\,|ds|
   \ =\ O(\tau^{-1/4})\ =\ o(\tau^{-1/2})
\]
\textup(numerically $\sqrt\tau$ times this bound $\to0$: $0.016,0.011,0.008,0.005,0.004,0.003$ over
$\tau\in[2\!\times\!10^{-2},5\!\times\!10^{-4}]$; \texttt{GAPU4\_numerator.py}\textup). Hence
$S_0-w\sin w=O(\tau^{-1/4})=o(w)$, so $S_0/(w\sin w)\to1$. This rests only on Lemma~\ref{lem:Bbounded}
\textup($|g_s|\le2$\textup) and Stirling --- exactly the inputs of Lemma~\ref{lem:T2abs}. We record the payoff.

\begin{corollary}[Numerator non-vanishing]\label{cor:numer}
For all sufficiently large $m$, $S_0(q_m)\ne0$ and $\Sigma_0(q_m)\ne0$, with
$S_0(q_m)\sim(-1)^{m+1}w_m$; hence all but finitely many zeros $\{q_m\}$ of $1-S_1$
\textup(resp.\ $1-\Sigma_1$\textup) are genuine simple poles of $G_0$ \textup(resp.\
$G^{T}_0$\textup). This is the input to Lemma~\ref{lem:numerator} and
Theorem~\ref{thm:main}; it is now \emph{unconditional}, on the same footing as
Lemma~\ref{lem:T2abs} \textup(Lemma~\ref{lem:Bbounded} and Stirling\textup), with no remaining
numerical hypothesis.
\end{corollary}

\begin{proof}
At a pole, $S_1(q_m)=1$, so by \eqref{eq:c1} $\cos w_m=O(\sqrt\tau_m)\to0$ and
$|\sin w_m|\to1$. By the unconditional absolute-contour bound above, $S_0(q_m)=w_m\sin w_m+o(w_m)$, whence
$|S_0(q_m)|\ge w_m\bigl(1-o(1)\bigr)\to\infty$; in particular $S_0(q_m)\ne0$ for all large $m$,
and the sign alternates with $\sin w_m$. The same argument
applies to $\Sigma_0$ \textup(leading term $w\sin w$ unchanged; subleading $+\tfrac1{18}\cos w$\textup).
\end{proof}

\noindent This removes the only numerical hypothesis previously attached to the relaxed
case: the non-vanishing $\Sigma_0(q_m)\ne0$ is no longer a spot-checked input but a
consequence of the same $w$-asymptotic that produces the poles. Corollary~\ref{cor:numer}
is checked in \texttt{transcendence\_verify\_ALL.py} (strict sign alternation and
$|\Sigma_0(q_m)|/w_m\to1$ monotonically over the first $33$ poles, before the
$q_m$-crowding exceeds working precision).

\subsection{An unconditional half of the leading term}
\label{ss:split}

The formal $\sqrt\tau$ term splits into a piece that is \emph{unconditionally} bounded,
uniformly in $w$, and a quantitatively smaller residue carrying the entire remaining gap.
First rewrite the error by swapping the order of summation in \eqref{eq:reduce}: since
$R_j(w)=\sum_{i>j}(-1)^{i}w^{2i}/(2i)!$ and $\sum_{j<i}\mu_j=1-\rho_{i-1}$,
\begin{equation}\label{eq:Etele}
  E:=S_1-(1-\cos w)=\sum_{i\ge1}(-1)^{i}\frac{w^{2i}}{(2i)!}\,(1-\rho_{i-1})
\end{equation}
(absolutely convergent, verified to $120$ digits). The domination of
Lemma~\ref{lem:dom} sharpens to $\rho_{i}/\rho_{i-1}\le e^{-\tau}$, hence
$\rho_{i-1}\le e^{-i\tau}$; writing $B_i:=-\log\rho_{i-1}-i\tau\ge0$ and
$W:=w\,e^{-\tau/2}$, the split $1-\rho_{i-1}=(1-e^{-i\tau})+e^{-i\tau}(1-e^{-B_i})$
inserted into \eqref{eq:Etele} (using $w^{2i}e^{-i\tau}=W^{2i}$) gives $E=T_1+T_2$ with
\begin{equation}\label{eq:T1T2}
  T_1=\sum_{i\ge1}(-1)^{i}\frac{w^{2i}}{(2i)!}(1-e^{-i\tau})=\cos w-\cos W,
  \qquad
  T_2=\sum_{i\ge1}(-1)^{i}\frac{W^{2i}}{(2i)!}\,(1-e^{-B_i}).
\end{equation}

\begin{proposition}[Unconditional uniform bound on $T_1$]\label{prop:T1}
$T_1=\cos w-\cos\!\bigl(w e^{-\tau/2}\bigr)$ exactly, and for all $\tau>0$
\[
  |T_1|\;\le\; w\bigl(1-e^{-\tau/2}\bigr)\;\le\;\sqrt{\tau/2},
\]
uniformly in $w$; consequently $T_1=-\tfrac1{\sqrt2}\sqrt\tau\,\sin w+O(\tau)$ uniformly.
\end{proposition}

\begin{proof}
The closed form is the evaluation of two convergent cosine series in \eqref{eq:T1T2}.
The bound is $|\cos w-\cos W|\le|w-W|=w(1-e^{-\tau/2})$ (mean value theorem), and
$1-e^{-\tau/2}\le\tau/2$, so $|T_1|\le w\tau/2=\sqrt{\tau/2}$. Expanding
$w(1-e^{-\tau/2})=\tfrac{w\tau}2+O(w\tau^2)=\tfrac1{\sqrt2}\sqrt\tau+O(\tau^{3/2})$ and
$\cos w-\cos(w-\tfrac{w\tau}2+\dots)=-\tfrac{w\tau}2\sin w+O((w\tau)^2)$ gives the stated
expansion; both error terms are uniform in $w$. \end{proof}

Thus $-\tfrac1{\sqrt2}$ --- the end-factor half of $c_1$ --- is established
\emph{unconditionally and uniformly}. The residue
$T_2=\sum_{j\ge0}c_j\,\widetilde R_j$, $c_j=e^{-B_j}(1-e^{-(\,\delta_j-\tau)})\in[0,\delta_j-\tau]$
with $\delta_j=-\log(\rho_j/\rho_{j-1})$ and $\widetilde R_j$ the cosine remainder at
argument $W$, is structurally identical to $E$ but with weights smaller by a factor
$O(\tau^2 j^2)$; numerically $T_2=+\tfrac{\sqrt2}{36}\sqrt\tau\sin w+O(\tau)$, completing
$c_1=-\tfrac1{\sqrt2}+\tfrac{\sqrt2}{36}=-\tfrac{17\sqrt2}{36}$. \textbf{The entire
remaining gap is the single bound $|T_2|\to0$ uniformly} --- met unconditionally below as the absolute contour
bound $|T_2|=O(\tau^{1/4})\to0$ (Lemma~\ref{lem:T2abs}), the sharp $O(\sqrt\tau)$ being the saddle refinement;
and the qualitative sign-straddle that suffices for Theorem~\ref{thm:main} needs only the much weaker
\emph{per-point} bound $|T_2|<1$ at the discrete phases $w=N\pi$.

\subsection{Equivalent reductions and the obstruction}
\label{ss:obstruction}

Two further rigorous reductions sharpen the target. From the integral form of the cosine
remainder, $R_j(w)=\tfrac{(-1)^{j+1}}{(2j)!}\int_0^w(w-s)^{2j}\sin s\,ds$ (one integration
by parts; boundary terms vanish), averaging against $\mu_j$ gives the exact convolution
\begin{equation}\label{eq:conv}
  E=\int_0^w \sin(s)\,K'(w-s)\,ds,\qquad
  K'(u)=\sum_{j\ge0}\mu_j\frac{(-1)^{j+1}u^{2j}}{(2j)!}\ \ (\text{entire, even}),
\end{equation}
the interchange justified by domination ($\sum_j\mu_j u^{2j}/(2j)!\le\cosh u$). Hence
$|E|\le w\sup_{[0,w]}|K'|$, eliminating the $s$-oscillation entirely; and
$K'(0)=\mu_0=1-g(2\tau)\le\tau$ rigorously (the $[1/1]$ Pad\'e bound
$(2-y)/(2+y)\le e^{-y}$). Moreover $|R_j(w)|\le \hat t_{j+1}=w^{2j+2}/(2j+2)!$ gives a
super-exponential tail: for $J_0\ge2w$, $|\sum_{j\ge J_0}\mu_j R_j|\le\hat t_{J_0+1}
\le(e/4)^{2J_0+2}=O(\tau^{\infty})$, so $E$ is a finite head of $O(w)$ terms up to a
negligible error. Lemma~\ref{lem:cos} is therefore \emph{equivalent} to either of the clean
self-contained estimates
\[
  \sup_{u\in[0,w]}|K'(u)|\le C\tau
  \qquad\Longleftrightarrow\qquad
  \|K\|_{L^1[0,w]}=O(\sqrt\tau)
  \qquad\Longleftrightarrow\qquad
  |T_2|=O(\sqrt\tau),
\]
for the explicit entire function $K'$ (equivalently $K(u)=\sum_j\mu_j(-1)^{j+1}u^{2j+1}/(2j+1)!$).

\begin{remark}[Why this is a genuine oscillatory estimate, not bookkeeping]\label{rem:knife}
The bound is a \emph{knife-edge}: the supremum of $|K'|$ on $[0,w]$ equals $\mu_0$ but is
approached from below by the interior lobe near $u\approx\pi$ with vanishing relative margin
$\sim 0.8\,\tau$. Hence no soft argument can close it: the weights $\mu_j$ provably fail
Hankel positivity (the Hankel matrices $[\mu_{i+k}]$ have genuinely negative eigenvalues,
persistent at $200$-digit precision), so $K'$ is \emph{not} the cosine transform of a
positive measure and the Bochner bound $|K'(u)|\le K'(0)$ is unavailable; and monotonicity,
contraction, and triangle/Abel bounds all fail (the partial sums of $K'$ overshoot like
$e^{u}$). The remaining estimate is genuinely a uniform-in-$\tau$ saddle/Poisson estimate of
an oscillatory sum in the coupled limit $\tau\to0$, $j\sim\tau^{-1/2}$, for which no
off-the-shelf difference-equation theorem applies: Wong--Li \cite{WongLi1992}, Wang--Wong
\cite{WangWong2003} and Costin--Costin \cite{CostinCostin1996} all require a \emph{fixed}
recurrence with $n\to\infty$, whereas here the turning point $k\sim w=\sqrt{2/\tau}$ recedes
\emph{jointly} with the index; and the false/partial-theta and Mellin harmonic-sum machineries
(\cite{BringmannMilas2015}; \cite{FGD1995}) do not apply (the $q\to1$ behaviour is carried by
the full Lambert-series corrections, not the leading $q$-powers, and $S_1$ is a sum of
products, not a harmonic sum). This is recorded as the precise frontier, established by a
five-route multi-agent assault with adversarial verification (all routes reduce to this one
inequality; none closes it).

A base-$q$ \emph{self-similarity} route is likewise dead. The catalytic dilation in the
Bousquet--M\'elou--Jehanne functional equation is $t\to q^2 t$ (the catalytic variable, i.e.\
the index $k$), \emph{not} a base substitution $q\to q^2$: telescoped it gives exactly the
$k$-recursion $\Sigma_k(q)=A_k(q)+C_k(q)\,\Sigma_{k+2}(q)$ at \emph{fixed} $q$ (verified to
$10^{-60}$), which couples different $k$ at the same $q$ and is already fully integrated into
the criticality $\Sigma_1(q)=1$. There is no residual base-$q$ self-similarity: numerically no
exact rational relation $\Sigma_1(q)=\Phi(q)+\Psi(q)\,\Sigma_1(q^2)$ exists (Pad\'e residuals
grow with the sample count at every fixed degree, and likewise for the quadratic ansatz);
structurally $q\to q^2$ sends the Lambert factors $1/(1-q^m)\to1/(1-q^{2m})$, so $\Sigma_1(q^2)$
lives in the even-index Lambert subring and cannot reproduce the odd-$m$ poles of $\Sigma_1(q)$;
and most decisively the cosine frequency $w=\sqrt{2/\tau}$ scales as $w\to w/\sqrt2$ under
$q\to q^2$, so the zero lattice $\{(n+\tfrac12)\pi\}$ is rescaled by the irrational $\sqrt2$ ---
a single zero's $q\mapsto\sqrt q$ orbit $\{w_0\,2^{m/2}\}$ meets the cosine-zero lattice at most
finitely often, while the $q\to q^2$ orbit $\{\tau,2\tau,4\tau,\dots\}$ runs toward $q=0$ (away
from the $q=1$ accumulation point) and is exponentially sparser (ratio $2$) than the actual
zeros (whose $\tau$-ratio $\to1$). Hence self-similarity cannot propagate zeros, and
Lemma~\ref{lem:cos} is not bypassable this way.
\end{remark}

\subsection{Resolution of the oscillatory core: model subtraction}
\label{ss:modelsub}

The knife-edge of Remark~\ref{rem:knife} is dissolved by \emph{subtracting} the
closed-form leading kernel and expanding the residual through an exact resummation
identity that performs the oscillatory cancellation for free. This reduces
Lemma~\ref{lem:cos} to a single non-oscillatory majorization (Lemma~\ref{lem:tail} below).

\paragraph{Geometric model and residual.} With the geometric weights
$\mu_j^{\mathrm{geo}}=e^{-j\tau}(1-e^{-\tau})$ \textup(i.e.\ $\rho_j^{\mathrm{geo}}
=e^{-(j+1)\tau}$\textup) the kernel resums in closed form,
\begin{equation}\label{eq:Kmodel}
  K'_{\mathrm{model}}(u):=\sum_{j\ge0}\mu_j^{\mathrm{geo}}\frac{(-1)^{j+1}u^{2j}}{(2j)!}
  =-(1-e^{-\tau})\cos\!\bigl(u\,e^{-\tau/2}\bigr),\qquad |K'_{\mathrm{model}}|\le\tau,
\end{equation}
and $\int_0^w\sin(s)\,K'_{\mathrm{model}}(w-s)\,ds=\cos w-\cos(we^{-\tau/2})=T_1$
(Proposition~\ref{prop:T1}). Put $\mathrm{corr}:=K'-K'_{\mathrm{model}}$, so $E=T_1+T_2$
with $T_2=\int_0^w\sin(s)\,\mathrm{corr}(w-s)\,ds$, and Lemma~\ref{lem:cos} follows from
\begin{equation}\label{eq:corrgoal}
  \sup_{u\in[0,w]}|\mathrm{corr}(u)|\le C\tau\quad(\text{any absolute }C)
  \ \Longrightarrow\ |T_2|\le w\,C\tau=O(\sqrt\tau).
\end{equation}
Subtraction removes the knife-edge: numerically $\sup|\mathrm{corr}|/\tau\to\tfrac16$
(stable: $0.171,0.154,0.162$ at $\tau=0.02,0.005,0.001$), the supremum migrating to the
boundary $u\approx w$ with a \emph{non-vanishing} margin --- so a non-sharp constant suffices.

\paragraph{Resummation engine and layers.} The exact identity
\begin{equation}\label{eq:engine}
  \sum_{j\ge0}j^{p}\,e^{-j\tau}\frac{(-1)^{j}u^{2j}}{(2j)!}
  =(-\partial_\tau)^{p}\cos\!\bigl(u\,e^{-\tau/2}\bigr)=:D_p(W),\qquad W:=u\,e^{-\tau/2},
\end{equation}
\textup(verified $p=0,\dots,3$; $D_{p+1}=\tfrac{W}{2}\partial_W D_p$\textup) turns each
oscillatory $j$-sum into a \emph{trigonometric polynomial in $W$ with no residual
cancellation}. Expanding $\rho_j=e^{-(j+1)\tau}e^{-R_j}$ \textup($R_j=-\log\rho_j-(j+1)\tau
=\tfrac19\tau^2 j^3+\cdots\ge0$, with $R_j\le0.47\sqrt\tau$ on the effective range $j\le w$\textup)
gives $K'=\sum_{m\ge0}L_m$, $L_0=K'_{\mathrm{model}}$, each $L_m$ a closed-form
combination of the $D_p$. This is exact: the layers reconstruct $K'$ to residual
$O(\tau^{M+1})$ \textup(verified to $5\times10^{-17}$ at $M=6$\textup).

\paragraph{The crux layer $L_1$ (rigorous).} Symbolically,
\begin{align*}
  L_1&=\tau^2\!\Bigl[\tfrac{W^3\sin W}{72}-\tfrac{W^2\cos W}{16}-\tfrac{W\sin W}{48}\Bigr]
   +\tau^2 e^{-\tau}\!\Bigl[-\tfrac{W^3\sin W}{72}+\tfrac{7W^2\cos W}{48}+\tfrac{17W\sin W}{48}-\tfrac{\cos W}{6}\Bigr].
\end{align*}
The dangerous $W^3\sin W$ term --- which alone would be $O(\tau^2(2/\tau)^{3/2})=O(\sqrt\tau)$,
fatal --- has coefficient $\tfrac{\tau^2}{72}(1-e^{-\tau})=O(\tau^3)$: the difference
structure $\mu_j=\rho_{j-1}-\rho_j$ cancels the top $W$-power. The surviving dominant term
is $\tfrac{\tau^2}{12}W^2\cos W$, bounded by $\tau/6$ since $W^2\le 2/\tau$; in full,
\begin{equation}\label{eq:L1bound}
  \sup_{[0,w]}|L_1|\ \le\ \frac{\tau}{6}+\frac{13\sqrt2}{36}\,\tau^{3/2}+\frac{\tau^2}{6}
  \qquad(\text{rigorous, exact coefficients; dominates }\sup|L_1|\text{ for all tested }\tau).
\end{equation}

\paragraph{The tail bound: Touchard/Poisson, not a uniform constant.} The naive hope ---
that $\sup_{[0,w]}|D_p|\le K(w/2)^p$ with $K$ \emph{absolute} --- is \emph{false}: the ratio
$\sup|D_p|/(w/2)^p$ grows like $2^{-p}\mathsf B_p$ (Bell numbers), e.g.\ $1.4,21,7.6\!\cdot\!10^3,
2.2\!\cdot\!10^7$ at $p=10,20,30,40$. The cancellation cannot be discarded; it is captured exactly
by a probabilistic identity. Writing $D_p(W)=2^{-p}\,\mathrm{Re}\bigl[e^{iW}T_p(iW)\bigr]$ with
$T_p(x)=\sum_r S(p,r)x^r$ the Touchard polynomial ($S(p,r)\ge0$ Stirling numbers of the second
kind), one has the \emph{rigorous} per-$p$ bound and the \emph{Poisson-moment} identity
\begin{equation}\label{eq:touchard}
  |D_p(W)|\le 2^{-p}T_p(W)\le 2^{-p}T_p(w)\quad(0\le W\le w),\qquad
  T_p(w)=\mathbb E[N^{p}],\ \ N\sim\mathrm{Poisson}(w),
\end{equation}
so for any polynomial $Q$, $\sum_p[j^p]Q\cdot 2^{-p}T_p(w)=\mathbb E\bigl[Q(N/2)\bigr]$ ---
dissolving the need for any uniform-in-$p$ constant.

\paragraph{R-control (rigorous).} With $\varphi(y)=\log\!\bigl(\sinh(y/2)/(y/2)\bigr)$ one has
$b_i=\varphi((2i{+}2)\tau)+\varphi((2i{+}1)\tau)-\varphi(\tau)$ and the elementary inequality
\begin{equation}\label{eq:phibound}
  0\le\varphi(y)\le \tfrac{y^2}{24}\qquad(y>0),
\end{equation}
proved from the Mittag--Leffler expansion $\coth t-\tfrac1t=\sum_{n\ge1}\tfrac{2t}{t^2+n^2\pi^2}$
together with $2\zeta(2)/\pi^2=\tfrac13$. Summing gives $0\le R_j\le\tau^2 C(j)$,
$C(j)=(j+1)(2j+3)(4j+5)/72$ (positive coefficients).

Expand $R_j=\sum_{n\ge1}\lambda_n\tau^{2n}P_n(j)$ ($P_n$ of degree $2n{+}1$ with nonnegative
coefficients, $\lambda_n=[y^{2n}]\varphi(y)$ of alternating sign, $\limsup|\lambda_n|^{1/2n}=1/2\pi$).
\emph{Caveat that fixes the proof:} the absolute series $\bar R(j)=\sum_n|\lambda_n|\tau^{2n}P_n(j)$
converges only for $j<\pi/\tau$ and \emph{diverges} beyond; since $N\sim\mathrm{Poisson}(w)$ has positive
mass on $\{N\ge2\pi/\tau\}$, $\mathbb E[\bar R(N/2)^m]=\infty$. The fix is a \emph{finite} truncation.

\begin{lemma}[Tail majorization]\label{lem:tail}
$\displaystyle\sum_{m\ge2}\sup_{u\in[0,w]}|L_m|=O(\tau)$.
\end{lemma}

\begin{proof}
Fix the order $N=2$ and split $R_j=R_j^{[2]}+S_j$, $R_j^{[2]}=\lambda_1\tau^2P_1(j)+\lambda_2\tau^4P_2(j)$
(a polynomial of degree $5$), $S_j=\sum_{n\ge3}\lambda_n\tau^{2n}P_n(j)$. Put $J_0=\lceil2w\rceil$.
\emph{(i) Outer tail $j\ge J_0$.} By $0\le R_j\le\tau^2C(j)$ and $W\le w$, with $2j\ge4w$ past the peak
$j^\ast\!\sim\!W/2\le w/2$ of $W^{2j}/(2j)!$, $\sum_{j\ge J_0}(\tau^2C(j))^m\,W^{2j}/(2j)!=O(\tau^\infty)$
(super-exponential factorial decay), uniformly in $m\le$ any fixed power.
\emph{(ii) Remainder $S$ on $j<J_0$.} $|S_j|\le\sum_{n\ge3}|\lambda_n|(2j\tau)^{2n}\le C(2J_0\tau)^{6}
=O(\tau^{5/2})$ on $j<J_0$ (since $2J_0\tau\le4w\tau=4\sqrt{2\tau}$), so in $R_j^m=\sum_k\binom mk
(R_j^{[2]})^{m-k}S_j^k$ every $k\ge1$ term is $O(\tau^{5/2})$-smaller and contributes $o(\tau)$ to the tail.
\emph{(iii) Main term $(R_j^{[2]})^m$.} This is a polynomial in $j$; the engine~\eqref{eq:engine} and the
triangle inequality over its (finitely many) coefficients with~\eqref{eq:touchard} give, with
$\bar R^{[2]}(j)=|\lambda_1|\tau^2P_1(j)+|\lambda_2|\tau^4P_2(j)$ a \emph{fixed} polynomial,
$\sup_{[0,w]}\bigl|\sum_{j<J_0}(R_j^{[2]})^m(-1)^{j+1}\tfrac{u^{2j}}{(2j)!}e^{-j\tau}\bigr|
\le\mathbb E\bigl[\bar R^{[2]}(N/2)^m\bigr]+O(\tau^\infty)$, both copies in $\mu_j^{(m)}$ handled the same
($e^{-\tau}\le1$; the shift $j\mapsto j-1$ replaces $P(j)$ by $P(j{+}1)$, still nonnegative).
\emph{(iv) Sum over $m$.} As $\bar R^{[2]}$ is a fixed polynomial, $\mathbb E[\bar R^{[2]}(N/2)^m]<\infty$, and
$\sum_{m\ge2}\tfrac{2}{m!}\mathbb E[\bar R^{[2]}(N/2)^m]=2\,\mathbb E\bigl[\Psi(\bar R^{[2]}(N/2))\bigr]$,
$\Psi(x)=e^x{-}1{-}x$. This is $O(\tau)$: on the bulk $N\le2w$, $\bar R^{[2]}(N/2)=O(\sqrt\tau)$ and
$\Psi\le\tfrac12\bar R^{[2]\,2}e^{\bar R^{[2]}}$; the Poisson tail $N>2w$ contributes
$O(\tau^\infty)$ because the polynomial $\bar R^{[2]}(N/2)$ grows only polynomially while
$\mathbb P(N=n)$ decays factorially ($\mathbb P(N>2w)\le e^{-w(2\ln2-1)}$). Combining (i)--(iv),
$\sum_{m\ge2}\sup|L_m|\le2\,\mathbb E[\Psi(\bar R^{[2]}(N/2))]+O(\tau^{3/2})=O(\tau)$; numerically
$\le0.0135\,\tau$ (and this bound dominates the true tail $\sup|\mathrm{corr}-L_1|$ with $10\times$ margin).
\end{proof}

\emph{A gap in the Touchard bound \textup(found by adversarial re-checking, and stated plainly\textup).}
The bound of step~(iv) is \emph{not} valid as written: $\bar R^{[2]}(N/2)$ is a degree-$5$ polynomial in
$N$, so $2\,\mathbb E[\Psi(\bar R^{[2]}(N/2))]=2\,\mathbb E[e^{\bar R^{[2]}(N/2)}-1-\bar R^{[2]}(N/2)]$
is \emph{infinite}: $\mathbb E[e^{c N^{5}}]=\sum_n e^{cn^{5}}w^{n}e^{-w}/n!$ diverges for every $c>0$
\textup(numerically, the partial sums sit near $1$ through the Poisson bulk and then explode --- $\sim
10^{975}$ by $n=400$ at $\tau=10^{-2}$\textup), because $e^{cn^{5}}$ ultimately outgrows the factorial
$1/n!$. The defect is that summing the per-$m$ \emph{absolute} Touchard bounds $|D_p|\le2^{-p}T_p(w)$
discards exactly the oscillatory cancellation that keeps the true series finite, and the Bell-number
growth of the Touchard polynomials then makes the $m$-sum diverge. So step~(iv) does \emph{not}
establish Lemma~\ref{lem:tail}. The quantity $\sup|\mathrm{corr}|$ \emph{is} genuinely $O(\tau)$ ---
$\mathrm{corr}$ is a bounded real function, with $\sup|\mathrm{corr}|/\tau=0.167$ at $\tau=10^{-4}$
to $120$ digits and the uniform-phase scan of \S\ref{ss:numerator} pinning $\sup_w|S_1-(1-\cos w)|/
\sqrt\tau$ at $|c_1|$ --- but an \emph{oscillation-preserving} \textup(saddle/steepest-descent\textup)
estimate is required to prove it, which we do not yet have. What \emph{is} rigorous is the leading order:
$T_1=\cos w-\cos(we^{-\tau/2})$ exactly with $|T_1|\le\sqrt{\tau/2}$, and $L_0,L_1$ \textup(the $\sqrt\tau$
coefficient via the $\vartheta^{3}$-resummation\textup). \emph{We therefore state Lemma~\ref{lem:cos}
as numerically certain but not yet proved}; consequently the \emph{ratio} transcendence below
\textup(Theorem~\ref{thm:main}: $G_0,G^{T}_0$, and $V$\textup) is conditional on it. The
\emph{individual blocks} $\Sigma_1,\Sigma_0,S_1,S_0$ remain transcendental \emph{unconditionally}
\textup(Theorem~\ref{thm:blocks}, P\'olya--Carlson, independent of Lemma~\ref{lem:cos}\textup), as
does the exact growth rate $\beta_2$ \textup(Euler squeeze\textup); these are untouched by the gap.
Note that Proposition~\ref{prop:signchanges} requires only that the error $\to0$ along the
\emph{extreme-phase} sequences $w=n\pi$ \textup(where the $\sqrt\tau\sin w$ term vanishes and the error
is the sharper $O(\tau)$\textup) --- a strictly weaker statement than the uniform bound, numerically
certain through $n\le64$ --- so closing transcendence needs only this extreme-phase decay, not the
full uniform Lemma~\ref{lem:cos}. \emph{This status is upgraded below}: the uniform decay $|T_2|=O(\tau^{1/4})\to0$ ---
all that $V$ needs (it forces $|E(m\pi)|<1$) --- is proved as an \emph{absolute contour bound}
(Lemma~\ref{lem:T2abs}), resting only on Lemma~\ref{lem:Bbounded} \textup(proved in closed form\textup) and
Stirling's asymptotics, with \emph{no} oscillatory-cancellation step. So $V$'s ratio transcendence
is unconditional modulo those textbook inputs, \emph{not} the open uniform Lemma~\ref{lem:cos}; $U$ adds only
parallel same-class steepest-descent estimates (closed at the same standard, \S\ref{sec:lifting}). \textup(The saddle route of \S\ref{ss:saddle} and
Remarks~\ref{rem:olverhyp}--\ref{rem:vdcred} additionally pins the sharp constant $\tfrac{\sqrt2}{36}$.\textup)

\subsection{The correct route: steepest descent}\label{ss:saddle}

The failure of the Touchard bound is not a failure of the \emph{result} but of the
\emph{method}: any estimate that takes absolute values term-by-term destroys the
oscillatory cancellation and meets the divergent majorant $\cosh W\sim e^{W}$. The
cancellation is recovered by deforming into the complex plane --- the standard
saddle-point/steepest-descent picture --- and this both identifies the leading term
in closed form and matches it to full numerical precision.

Write the residual as the value of an entire function on the imaginary axis:
\[
  T_2 \;=\; G(iW),\qquad G(z)=\sum_{i\ge1} g_i\,\frac{z^{2i}}{(2i)!},\qquad
  g_i=1-e^{-B_i}\in[0,1),
\]
since $(iW)^{2i}=(-1)^iW^{2i}$. On the real axis $G$ has exponential type $1$
($|G(x)|\le\cosh x$), so the imaginary axis is the \emph{subdominant} direction
(the indicator $h(\pi/2)=0$, as for $\cosh(iW)=\cos W$); the magnitude of $G(iW)$ is
governed not by the type but by the saddle. Treating $i$ as continuous, the summand
$(-1)^i W^{2i}/(2i)!$ has its stationary point (with $(-1)^i=e^{i\pi i}$ and Stirling)
at the \emph{complex} node
\[
  s^\ast=\tfrac{i}{2}\,W .
\]
Two facts make this decisive. First, $B_s$ continues analytically to $s^\ast$ and is
\emph{small} there: its $\tau$-series $B_s=\sum_{n\ge1}f_n\tau^{2n}Q_n(s)$
($\deg Q_n=2n{+}1$) converges at $s^\ast$ because $\tau\,(W/2)^2=\tfrac12$ lies inside
the radius (set by the $2\pi$-radius of $\phi$), and its leading term gives
\[
  B_{s^\ast}=\tfrac{\tau^2}{9}\Bigl(\tfrac{i}{2}W\Bigr)^{3}+O(\tau)
  =-\,i\,\frac{\tau^2W^3}{72}+O(\tau)
  =-\,i\,\frac{\sqrt2}{36}\sqrt\tau+O(\tau),
\]
using $\tau^2W^3=2\sqrt2\,\sqrt\tau$. Second, $g_{s^\ast}=1-e^{-B_{s^\ast}}=B_{s^\ast}+O(\tau)$.
The steepest-descent contribution of the node $s^\ast$ (with its conjugate $-s^\ast$,
assembling the real part) is then
\begin{equation}\label{eq:saddle}
  T_2=\operatorname{Re}\!\bigl[g_{s^\ast}e^{iW}\bigr]+O(\tau)
     =\frac{\sqrt2}{36}\,\sqrt\tau\,\sin w+O(\tau),
\end{equation}
the leading coefficient agreeing with~\eqref{eq:c1} \textup(together with the
unconditional $T_1$ this reassembles $c_1$\textup). Identity~\eqref{eq:saddle} is
confirmed in \texttt{saddle\_point\_lemcos.py}: the ratio $T_2/\operatorname{Re}[B_{s^\ast}e^{iW}]\to1$
as $\tau\to0$ \textup($1.0005$ at $\tau=4\times10^{-4}$\textup), with no free constant.

The correct contour is the \emph{Lindel\"of} steepest-descent path: with
$h(s)=W^{2s}g_s/\Gamma(2s+1)$ (so $T_2=\sum_{i\ge1}(-1)^i h(i)$), one has the exact
$T_2=\tfrac{1}{2\pi i}\oint_C h(s)\,\pi/\!\sin(\pi s)\,ds$, $C$ encircling the positive
integers, and $h$ is \emph{analytic in} $\{\operatorname{Re}s\ge0\}$ (the poles of
$B_s$ lie at $\operatorname{Re}s<0$), so $C$ deforms to the steepest path through
$s^\ast=\tfrac i2 W$ without obstruction. Here $B_s$ is known in closed form:
$B_s=\sum_{k\ge1}\bigl[T_{2,k}(s)+T_{1,k}(s)-s\log(1+(\tfrac{\tau}{2\pi k})^2)\bigr]$,
$T_{c,k}(s)=2s\log\tfrac{\tau}{\pi k}+\log\Gamma(s{+}\tfrac c2{+}\tfrac{i\pi k}{\tau})+\log\Gamma(s{+}\tfrac c2{-}\tfrac{i\pi k}{\tau})-\log\Gamma(\tfrac c2{\pm}\tfrac{i\pi k}{\tau})$,
from the product $\sinh(y/2)/(y/2)=\prod_k(1+(\tfrac{y}{2\pi k})^2)$
\textup(verified against the form factors to full precision\textup).
Two straight-contour shortcuts are \emph{provably} unavailable: the Abel--Plana /
vertical-Lindel\"of forms force the full imaginary axis, where $\operatorname{Re}B_{iy}\to-\infty$
for $y\gg W$, making $g_{iy}$ blow up and the integral diverge --- the curved path,
which stays at $\operatorname{Im}s\le W/2$ (where $B$ is small), is genuinely necessary.
On the saddle path the magnitudes assemble correctly: $|h(s^\ast)/\!\sin(\pi s^\ast)|
=|g_{s^\ast}|\,|\Gamma(1{+}iW)|^{-1}/\sinh(\tfrac{\pi W}{2})$, and the two factors
$e^{\pi W/2}$ \textup(from $\Gamma(1{+}iW)^{-1}$ and from $\sinh\tfrac{\pi W}{2}$\textup)
\emph{cancel}, leaving the saddle contribution $|g_{s^\ast}|\cdot O(1)\cdot\sqrt{2\pi/|F''(s^\ast)|}$;
numerically this magnitude divided by $\sqrt\tau$ converges to $\tfrac{\sqrt2}{36}=0.03928\ldots$
\textup(verified at dps$\ge50$, both directly and symbolically: $f_1=\tfrac1{24}$, $Q_1(s){\sim}\tfrac83 s^3$,
so $B_s{\sim}\tfrac19\tau^2 s^3$ and $B_{s^\ast}=-i\tfrac{\sqrt2}{36}\sqrt\tau$ on shell\textup), the
coefficient of \eqref{eq:saddle} with no free constant. What is firmly established is the
\emph{leading order}: the node $s^\ast$ is forced, $B_{s^\ast}=O(\sqrt\tau)$, $h$ is analytic in
$\{\operatorname{Re}s\ge0\}$ so the deformation is legal \textup(the rectangular-contour residue sum
reproduces $\sum(-1)^n h(n)$ to $10^{-16}$\textup), and a \emph{convergent} bent steepest-descent
path through $s^\ast$ exists. \emph{We do not, however, claim a complete error bound.} Adversarial
re-checking shows the naive routes fail: the standard real-axis representation diverges \textup(above\textup),
so a van der Corput / Olver theorem cannot be applied to it; bounding a truncated-integral tail by
the convergent series tail is a category error \textup(the two differ by $O(1)$, not $o(1)$, since the
integral amplitude \emph{grows} toward $y=\pi/\tau$\textup); and $\tfrac{\sqrt2}{36}$ is the leading
coefficient \emph{approached from above}, not a finite-$\tau$ upper bound. What survives is:
$\sup_w|T_2|/\sqrt\tau\to\tfrac{\sqrt2}{36}$ as $\tau\to0$, and the empirical inequality
$|T_2|\le0.06\sqrt\tau$ on $(0,0.3]$ \textup(global $\sup\,|T_2|/\sqrt\tau=0.0538$ at $\tau\approx0.224$\textup).
Thus the leading asymptotic is rigorous, and the right tool for a rigorous $|T_2|=O(\sqrt\tau)$ bound is the
van der Corput second-derivative lemma on the stationary-phase form, \emph{not} a saddle-dominated descent
(Remark~\ref{rem:olverhyp}); the one written-out step is the half-power-sharper extreme-phase bound
$|T_2(m\pi)|=O(\tau)$, the leading term subtracted. We state Lemma~\ref{lem:cos} accordingly: leading order
and $O(\sqrt\tau)$ proved, the extreme-phase $O(\tau)$ sharpening the remaining step.

\begin{lemma}[Cosine asymptotics; reduced to one oscillatory-average bound]\label{lem:cos}
As $q\to1^{-}$, with $\tau=-\ln q$ and $w=\sqrt{2/\tau}$,
\begin{equation}\label{eq:cosasymp}
  S_1(q)=1-\cos\!\sqrt{\tfrac{2}{-\ln q}}\;+\;O\!\bigl(\sqrt{-\ln q}\,\bigr),
  \qquad
  \Sigma_1(q)=1-\cos\!\sqrt{\tfrac{2}{-\ln q}}\;+\;O\!\bigl(\sqrt{-\ln q}\,\bigr),
\end{equation}
the error bounds being uniform in the phase $w$. Equivalently, in $\varepsilon=1-q$,
$S_1(q)=1-\cos\sqrt{2/\varepsilon}+O(\sqrt{\varepsilon})$.
\end{lemma}

\noindent\emph{Numerical certification.} Equation \eqref{eq:cosasymp} is
confirmed to $8$+ significant figures across the scales
$\varepsilon=10^{-3},\dots,10^{-8}$ and across all phases: the period-maximised
error satisfies $\sup_w\lvert S_1-(1-\cos w)\rvert\le 0.67\sqrt{\tau}$ and
$\sup_w\lvert \Sigma_1-(1-\cos w)\rvert\le 0.04\sqrt{\tau}$ on $\tau\in(0,10^{-3}]$
(monotone in the sampled range), and at the extreme phases the deviation is the
sharper $O(\tau)$: $S_1\big|_{w=2n\pi}=0.0980\,\tau+o(\tau)\to0$ and
$2-S_1\big|_{w=(2n+1)\pi}=0.0980\,\tau+o(\tau)\to0$
(\texttt{transcendence\_verify\_ALL.py}, \texttt{rate\_extreme.py}). We use only
the qualitative consequence below.

\begin{lemma}[$\operatorname{Re}B_s$ is bounded below on the contour strip]\label{lem:Bbounded}
For every $\tau\in(0,2\pi^2)$, $\operatorname{Re}B_s$ is bounded below on the strip
$\mathcal S=\{\,0\le\operatorname{Im}s\le W/2,\ \operatorname{Re}s\ge0\,\}$, with
\[
   \inf_{\mathcal S}\operatorname{Re}B_s=-\tfrac{\sqrt2}{18}\sqrt\tau+O(\tau)\ \longrightarrow\ 0,
\]
attained at the \emph{diagonal} node $s=\tfrac W2(1+i)$ \textup(not the corner $\sigma=0$\textup). Hence
$A=1-e^{-B_s}$ is bounded on $\mathcal S$ \textup(by $1+e^{(\sqrt2/18)\sqrt\tau}\to2$\textup); as the
operative contour lies in $\mathcal S$ \textup(\S\ref{ss:saddle}\textup), $V_\Gamma(A)<\infty$.
\end{lemma}

\begin{proof}
Write $s=\sigma+it\in\mathcal S$ and use the $\Gamma$-function form
\[
  B_s=\sum_{k\ge1}\sum_{c\in\{1,2\}}\!\Bigl[2s\log\tfrac{\tau}{\pi k}
  +\log\Gamma\bigl(s{+}\tfrac c2{+}\tfrac{i\pi k}{\tau}\bigr)
  +\log\Gamma\bigl(s{+}\tfrac c2{-}\tfrac{i\pi k}{\tau}\bigr)
  -\log\Gamma\bigl(\tfrac c2{+}\tfrac{i\pi k}{\tau}\bigr)
  -\log\Gamma\bigl(\tfrac c2{-}\tfrac{i\pi k}{\tau}\bigr)\Bigr]
  -\sum_{k\ge1}s\log\!\bigl(1{+}(\tfrac{\tau}{2\pi k})^2\bigr),
\]
convergent for $\operatorname{Re}s\ge0$ (its poles lie in $\operatorname{Re}s<0$). The crux is the
elementary inequality
\[
  \tfrac W2=\tfrac1{\sqrt2}\,\tau^{-1/2}<\pi\,\tau^{-1}=\tfrac\pi\tau
  \qquad(\text{true whenever }\tau<2\pi^2),
\]
so on $\mathcal S$, $t\le W/2<\pi/\tau\le\pi k/\tau$ for all $k\ge1$: there is \emph{no resonance}, and
each imaginary part $t\pm\tfrac{\pi k}\tau$ has modulus $\ge\tfrac{\pi k}\tau-t\ge\tfrac\pi\tau-\tfrac W2>0$.
By Stirling $\operatorname{Re}\log\Gamma(x{+}iY)=(x{-}\tfrac12)\log|Y|-\tfrac\pi2|Y|+\tfrac12\log2\pi+O(|Y|^{-1})$,
uniformly for $x$ in compacta: the pair $Y=t{+}\tfrac{\pi k}\tau$, $Y=t{-}\tfrac{\pi k}\tau$ contributes
$-\tfrac\pi2(t{+}\tfrac{\pi k}\tau)-\tfrac\pi2(\tfrac{\pi k}\tau{-}t)=-\tfrac{\pi^2k}\tau$ (the $t$'s cancel),
\emph{exactly} cancelling the $-\tfrac{\pi^2k}\tau$ of the boundary pair $Y=\pm\tfrac{\pi k}\tau$; the
$(x{-}\tfrac12)\log|Y|$ parts cancel against the prefactor $2\sigma\log\tfrac\tau{\pi k}$. What survives is
$\sum_k O\!\bigl((t\tau/\pi k)^2\bigr)$, a convergent series ($t\le W/2<\pi/\tau$): there is \emph{no
resonance} and $\operatorname{Re}B_s$ is finite throughout $\mathcal S$.

For the explicit lower bound, expand in $\tau$. From the form-factor representation
$B_s=\sum_{m=1}^{2s}\varphi(m\tau)-s\,\varphi(\tau)$ with
$\varphi(y)=\log\frac{\sinh(y/2)}{y/2}=\sum_{n\ge1}\varphi_n y^{2n}$,
$\varphi_n=\frac{(-1)^{n+1}\zeta(2n)}{n(2\pi)^{2n}}$, and the power sum $P_n(M)=\sum_{m=1}^M m^{2n}$
\textup(Faulhaber, $\deg P_n=2n{+}1$\textup), term-by-term summation gives the \emph{exact} series
\[
   B_s=\sum_{n\ge1}\varphi_n\,\tau^{2n}\bigl(P_n(2s)-s\bigr)\qquad(|s|<\pi/\tau),
\]
verified against the $\Gamma$-form above to its truncation (\texttt{ADV\_Bound1\_tauseries.py}). Successive
terms shrink by $O\bigl((|s|\tau/\pi)^2\bigr)=O(\tau)$ at the scale $|s|\asymp\tau^{-1/2}$ that matters, so the
tail $\sum_{n\ge2}$ is $O(\tau^{3/2})$ and the bound is set by $n=1$:
$\varphi_1\tau^2(P_1(2s)-s)=\tfrac1{36}\tau^2(4s^3+3s^2-s)$
\textup($\varphi_1=\tfrac1{24}$, $P_1(M)=\tfrac{M(M{+}1)(2M{+}1)}6$\textup), with real part
$\tfrac1{36}\tau^2\bigl(4(\sigma^3{-}3\sigma t^2)+3(\sigma^2{-}t^2)-\sigma\bigr)$. The dominant cubic
$\sigma^3-3\sigma t^2$ is minimized on $\mathcal S$ at $\sigma=t=W/2$ (where $\partial_\sigma=3\sigma^2-3t^2=0$,
$t$ maximal) at value $-W^3/4$; since $\tau^2W^3=2\sqrt2\sqrt\tau$,
\[
   \operatorname{Re}B_s\ \ge\ -\tfrac{\sqrt2}{18}\sqrt\tau+O(\tau)\ \longrightarrow\ 0,
\]
the minimum sitting on the top edge $\operatorname{Im}s=W/2$ near the diagonal $\sigma\approx W/2$ \textup(an
$O(\tau)$ shift, $\sigma^\ast/W\approx0.46$--$0.49$, toward the cubic--quadratic balance\textup) --- \emph{not}
the corner $\sigma=0$; $\operatorname{Re}B_s$ is
\emph{not} monotone in $\sigma$ (it dips to the minimum near $\sigma=W/2$, slope $-\tau/6$, before the real-axis
growth of $\log\Gamma(2s{+}1)$ sends it to $+\infty$). On the imaginary axis $s=iy$ --- where the amplitude $A$
of \S\ref{ss:saddle} lives --- the real part has the closed form
\[
   \operatorname{Re}B_{iy}=-\tfrac12\log\frac{y\tau}{\sin(y\tau)}=-\tfrac\tau{24}+O(\tau^2)\qquad(0<y\tau<\pi),
\]
from the generating identity $\sum_{n\ge1}\tfrac{\zeta(2n)}n z^{2n}=-\log\tfrac{\sin\pi z}{\pi z}$ applied to
the even part of the series (the odd Bernoulli numbers $B_{2k+1}=0$ killing every lower Faulhaber term in
$\operatorname{Re}P_n(2iy)$); verified to $8$ digits. This is $\ge-\tfrac{\sqrt2}{18}\sqrt\tau$ a fortiori, so
$\operatorname{Re}B_s\ge-\tfrac{\sqrt2}{18}\sqrt\tau\to0$ on $\mathcal S$. The cancellation breaks
only when some $\pi k/\tau<t$, i.e.\ $t>\pi/\tau$: there the resonant $k\le t\tau/\pi$ contribute
$-\pi t+\pi^2k/\tau<0$, summing to $\sim-\tfrac12t^2\tau\to-\infty$ — the imaginary-axis divergence — but
that locus $\operatorname{Im}s>\pi/\tau$ is disjoint from $\mathcal S$.
\end{proof}

\noindent\emph{Numerically confirmed} (\texttt{ADV\_Bbounded\_verify.py}, dps$\ge45$): the strip minimum
sits at the diagonal $s=\tfrac W2(1+i)$ with $\operatorname{Re}B_s/\sqrt\tau\to-\tfrac{\sqrt2}{18}=-0.07857$
($-0.0794,-0.0789,-0.0787$ at $\tau=0.05,0.02,0.01$, Richardson exponent $\approx1$); along the top edge the
value dips to this minimum near $\sigma=W/2$ and only then rises (\emph{not} monotone), while the $\sigma=0$
edge carries the milder $-\tfrac1{24}\tau$; and the resonance disaster is off-strip --- at
$\operatorname{Im}s=\pi/\tau$, $\operatorname{Re}B_s=-8.2$; at $\operatorname{Im}s=80$, $-110$
(\texttt{lemcos\_Bstrip.py}).

\begin{lemma}[Absolute contour bound: $|T_2|=O(\tau^{1/4})\to0$, no oscillatory cancellation]\label{lem:T2abs}
Let $h(s)=g_s\,W^{2s}/\Gamma(2s+1)$, $g_s=1-e^{-B_s}$, so that $T_2=\sum_{n\ge1}(-1)^n h(n)$. By the residue
theorem \textup($\operatorname{Res}_{s=n}\tfrac{\pi}{\sin\pi s}=(-1)^n$\textup),
\[
   T_2=\frac{1}{2\pi i}\oint_{\partial R}h(s)\,\frac{\pi}{\sin\pi s}\,ds,\qquad
   R=\bigl\{\operatorname{Re}s\ge\tfrac12,\ |\operatorname{Im}s|\le W/2\bigr\},
\]
legal because $h$ is analytic on $\{\operatorname{Re}s>0\}$ \textup(the poles of $B_s$ lie in
$\operatorname{Re}s<0$\textup) and decays factorially as $\operatorname{Re}s\to+\infty$; $\partial R$ encircles
exactly the positive integers. Hence
$|T_2|\le\frac1{2\pi}\oint_{\partial R}|h(s)|\,\bigl|\tfrac\pi{\sin\pi s}\bigr|\,|ds|=:A$, and we bound $A$
\emph{in modulus} from two rigorous facts about the $\tau$-series $B_s=\sum_{n\ge1}\varphi_n\tau^{2n}(P_n(2s)-s)$:
\begin{itemize}[leftmargin=1.7em,itemsep=0.1em]
\item[\textup{(i)}] \emph{global amplitude.} $\operatorname{Re}B_s\ge-\tfrac{\sqrt2}{18}\sqrt\tau\,(1+o(1))$ on
$\partial R$ \textup(Lemma~\ref{lem:Bbounded}: there is no resonance, $|\operatorname{Im}s|\le W/2<\pi/\tau$, and
the dominant cubic of $\operatorname{Re}B_s$ rises back to $+\infty$ for $\operatorname{Re}s\gtrsim\tfrac{\sqrt3}2W$,
so the lower bound holds along the \emph{entire} horizontal sides\textup), whence
$|g_s|=|1-e^{-B_s}|\le1+e^{(\sqrt2/18)\sqrt\tau}\le2$ everywhere on $\partial R$;
\item[\textup{(ii)}] \emph{inner cubic.} on $\{|s|\le W\}$ the series is dominated by its $n=1$ term
$\tfrac1{24}\tau^2(P_1(2s)-s)$ (successive terms shrink by $(|s|\tau/\pi)^2\le\tau/\pi^2$), giving the rigorous
$|B_s|\le K\tau^2|s|^3$ with $K=\tfrac18(1+o(1))$; there $|B_s|\le C\sqrt\tau\le\log2$, so
$|g_s|\le|B_s|e^{|B_s|}\le2K\tau^2|s|^3$.
\end{itemize}
Write $A=A_{\mathrm{horiz}}+A_{\mathrm{left}}$.

\emph{Horizontal sides} $\operatorname{Im}s=\pm W/2$. Here $|{\sin\pi s}|^2=\sin^2\pi\sigma+\sinh^2\tfrac{\pi W}2
\ge\sinh^2\tfrac{\pi W}2$, so $\bigl|\tfrac\pi{\sin\pi s}\bigr|\le\pi/\sinh\tfrac{\pi W}2\le2.01\,\pi\,e^{-\pi W/2}$,
and this $e^{-\pi W/2}$ \emph{cancels} the $e^{+\pi W/2}$ inside $1/|\Gamma(2\sigma+1+iW)|$: the divergent real-axis
majorant $\cosh W\sim e^{W}$ is gone, so passing to absolute values is harmless. With $|g_s|\le C\sqrt\tau$ on the
inner range $\tfrac12\le\sigma\le\tfrac{\sqrt3}2W$ and the factorial decay of $W^{2\sigma}/|\Gamma(2\sigma+1+iW)|$
killing the tail $\sigma>\tfrac{\sqrt3}2W$ (where $|g_s|\le2$) up to $O(\tau^P)$ for every $P$, the crude
majorant gives $A_{\mathrm{horiz}}=O(\tau^{1/4})$ --- of the same order as the left side, hence harmless;
the sharp size is $A_{\mathrm{horiz}}=0.06\,\sqrt\tau\,(1+o(1))$ numerically (\texttt{GAPV\_peredge.py}), the
extra half-power requiring the saddle of \S\ref{ss:saddle}.

\emph{Left side} $\operatorname{Re}s=\tfrac12$, $s=\tfrac12+it$ --- the bottleneck. Here
$\bigl|\tfrac\pi{\sin\pi s}\bigr|=\pi/\cosh\pi t$ and $|W^{2s}/\Gamma(2s{+}1)|=W/|\Gamma(2+2it)|$, while (ii) gives
$|g_s|\le 2K\tau^2(\tfrac14+t^2)^{3/2}$. By Stirling $|\Gamma(2+2it)|\sim\sqrt{2\pi}(2t)^{3/2}e^{-\pi t}$ and
$\cosh\pi t\sim\tfrac12e^{\pi t}$, so $W/(|\Gamma(2+2it)|\cosh\pi t)\sim\tfrac{\sqrt\pi}2\,W\,t^{-3/2}$ and the
integrand is $\lesssim\tau^2W\,t^{3/2}$ (the $e^{\pm\pi t}$ cancel). Hence the \emph{growing} cubic amplitude on the
vertical return controls the size:
\[
   A_{\mathrm{left}}=\tfrac1\pi\!\int_0^{W/2}\!\!|g|\,\tfrac{W}{|\Gamma(2+2it)|}\,\tfrac\pi{\cosh\pi t}\,dt
   \ \lesssim\ \tau^2W\!\int_0^{W/2}\!\! t^{3/2}\,dt\ \asymp\ \tau^2W\cdot W^{5/2}=\tau^2W^{7/2}\asymp\tau^{1/4},
\]
since $W=\sqrt{2/\tau}\,e^{-\tau/2}$ gives $\tau^2W^{7/2}=2^{7/4}\tau^{1/4}$. Therefore
\[
   |T_2|\ \le\ A\ =\ \underbrace{O(\sqrt\tau)}_{\text{horizontal}}\ +\ \underbrace{O(\tau^{1/4})}_{\text{left}}
   \ =\ O(\tau^{1/4})\ \longrightarrow\ 0 .
\]
\textup(Numerically $A_{\mathrm{left}}=0.00746\,\tau^{1/4}(1+o(1))$, fitted exponent $0.257$ over
$\tau\in[10^{-2},10^{-5}]$, \texttt{GAPV\_exponent.py}; and $A\ge|T_2|$ by construction.\textup)
\end{lemma}

\noindent This is the clean route to exactly what $V$ needs. It rests only on Lemma~\ref{lem:Bbounded}
\textup(proved in closed form\textup) and Stirling's asymptotics for $|\Gamma(2s+1)|$: \emph{no}
oscillatory-cancellation step, \emph{no} steepest-descent or van der Corput estimate, \emph{no} Poisson alias
bound --- only the residue theorem and two elementary majorants. The price of discarding all phase information
is the crude exponent: the absolute bound gives $|T_2|=O(\tau^{1/4})$, weaker than the true $O(\sqrt\tau)$, the
loss being the cubic growth of $|g_s|$ along the vertical return $\operatorname{Re}s=\tfrac12$. This is
immaterial, because what $V$ consumes is only \emph{decay}: the absolute bound gives
$|E|\le\sqrt{\tau/2}+|T_2|=O(\tau^{1/4})\to0$, hence $|E(m\pi)|<1$ for all large $m$, which is exactly the
hypothesis of Proposition~\ref{prop:signchanges}. The saddle analysis of \S\ref{ss:saddle} and
Remarks~\ref{rem:olverhyp}--\ref{rem:vdcred} stays valid and supplies the \emph{sharp} rate $O(\sqrt\tau)$ with
constant $\tfrac{\sqrt2}{36}$, but is no longer needed for transcendence.

\begin{lemma}[Extreme-phase error; the form pole accumulation actually needs]\label{lem:extremephase}
Write $E(w)=\Sigma_1(e^{-2/w^2})-(1-\cos w)$ and $E^{\mathrm{b}}(w)=S_1(e^{-2/w^2})-(1-\cos w)$. Along
the extreme-phase sequence $w=m\pi$ ($m\in\mathbb N$, $\tau_m=2/(m\pi)^2\to0$),
\[
   E(m\pi)=O(\tau_m),\qquad E^{\mathrm{b}}(m\pi)=O(\tau_m),
   \qquad\text{in particular }|E(m\pi)|,\,|E^{\mathrm{b}}(m\pi)|<1\text{ for all large }m.
\]
This is one half-power sharper than the generic $O(\sqrt\tau)$ of Lemma~\ref{lem:cos}, \emph{because}
the leading saddle term $\tfrac{\sqrt2}{36}\sqrt\tau\,\sin w$ of \eqref{eq:saddle} vanishes identically
at $w=m\pi$ ($\sin m\pi=0$).
\end{lemma}

\begin{proof}
Write $E=T_1+T_2$ as in \S\ref{ss:saddle}, with $|T_1|\le\sqrt{\tau/2}$ unconditional; it suffices to
bound $T_2(m\pi)$. From \S\ref{ss:saddle},
$T_2=\tfrac1{2\pi i}\oint_C h(s)\,\dfrac{\pi}{\sin\pi s}\,ds$, $h(s)=W^{2s}g_s/\Gamma(2s{+}1)$,
$g_s=1-e^{-B_s}$, $C$ encircling the positive integers, and $h$ is analytic in
$\{\operatorname{Re}s\ge0\}$, so $C$ deforms to the operative stationary-phase contour $\Gamma$
($\operatorname{Im}s=W/2$; Remark~\ref{rem:olverhyp}) without crossing a singularity. Set $\Phi(s)=\log\!\bigl[
W^{2s}/(\Gamma(2s{+}1)\sin\pi s)\bigr]$ and amplitude $A(s)=g_s$, so $T_2=\tfrac1{2i}\int_\Gamma
e^{\Phi(s)}A(s)\,ds$. Three properties hold on the curved contour, all from \S\ref{ss:saddle}:
\begin{itemize}[leftmargin=1.4em,itemsep=0.1em]
\item[\textup{(a)}] nondegenerate stationary point, $\Phi''=-4/W\ne0$ on the real section (the complex saddle $s^\ast=-\tfrac14+\tfrac i2W$ carries $\Phi''=+4i/W$; Remark~\ref{rem:olverhyp});
\item[\textup{(b)}] on $\Gamma$, which stays in the strip $\mathcal S$ ($\operatorname{Im}s\le W/2$), the
$\tau$-series of $B_s$ converges ($\tau(W/2)^2=\tfrac12$, inside its radius), so $B_s$ is analytic with
$|B_s|\le C\sqrt\tau$ and $A=B_s+O(B_s^2)$ is analytic near the saddle; and by
Lemma~\ref{lem:Bbounded}, $\operatorname{Re}B_s$ is bounded below on $\mathcal S$, so $A$ is
\emph{bounded} on $\Gamma$ and of \emph{finite total variation} $V_\Gamma(A)<\infty$. This is exactly
where the \emph{real}-axis representation fails ($\operatorname{Re}B_{iy}\to-\infty$ at the resonance
$\operatorname{Im}s\approx\pi/\tau\gg W/2$, $A$ unbounded) and why the curved path, confined to
$\mathcal S$ below the resonance, is necessary;
\item[\textup{(c)}] the two factors $e^{\pi W/2}$, from $\Gamma(1{+}iW)^{-1}$ and from $\sinh(\pi W/2)$,
cancel, so the prefactor multiplying $A$ on $\Gamma$ is $O(1)$.
\end{itemize}
The leading $\sqrt\tau$ envelope of \eqref{eq:saddle}, $\tfrac{\sqrt2}{36}\sqrt\tau\,\sin w$, \emph{vanishes
identically} at an extreme phase $w=m\pi$ ($\sin m\pi=0$). The deviation is therefore the \emph{next}
stationary-phase order,
\[
   T_2(m\pi)=R(m\pi),\qquad |R(m\pi)|\le K\,V_\Gamma(A)\,\tau=O(\tau),
\]
with $K$ absolute and $V_\Gamma(A)<\infty$ by Lemma~\ref{lem:Bbounded}: once the $\sqrt\tau$ envelope is gone,
finiteness of the amplitude variation, not smallness, is all that enters. This $O(\tau)$ remainder is the van
der Corput stationary-phase estimate \emph{with the leading term subtracted} --- the single written-out step
flagged in Remark~\ref{rem:olverhyp}, \emph{not} a saddle-dominance step. The identical argument with $S_1$ in
place of $\Sigma_1$ gives $E^{\mathrm{b}}(m\pi)=O(\tau)$. Hence $|E(m\pi)|\le\sqrt{\tau/2}+O(\tau)\to0$, so
$<1$ for all large $m$, as claimed.
\emph{(}Numerically $-E(m\pi)/\tau\to0.1242\ldots$ over $m\le62$, far inside $|E|<1$, confirming the
bound.\emph{)}
\end{proof}

\begin{remark}[Stationary phase, not saddle-dominance: what the contour trace shows]\label{rem:olverhyp}
A direct numerical trace of the descent geometry (\texttt{ADV\_descent\_trace.py}) sharpens
\S\ref{ss:saddle} and is worth stating plainly, as it corrects the natural first guess. The complex saddle
of $\Phi(s)=2s\log W-\log\Gamma(2s{+}1)-\log\sin\pi s$ sits at $s^\ast=-\tfrac14+i\bigl(\tfrac W2+O(\sqrt\tau)\bigr)$
--- the $-\log\sin\pi s$ term shifts it off the imaginary axis --- with $\Phi''(s^\ast)=+4i/W$ \emph{purely
imaginary} (descent tangents at $\pm45^\circ$, not horizontal). Two facts follow: \emph{(i)} the \emph{literal}
constant-phase descent path through $s^\ast$ does \emph{not} remain in $\mathcal S$ (one branch climbs to
$\operatorname{Im}s\approx\tfrac32\!\cdot\!\tfrac W2>W/2$ and curls back, the other runs to $\operatorname{Re}s<0$),
so a saddle-\emph{dominated} steepest-descent contour cannot be cited; \emph{(ii)} on the operative horizontal
contour $\operatorname{Im}s=W/2$ the saddle does \emph{not} dominate $|h|$ (between integer poles
$1/|\!\sin\pi s|$ spikes far above the saddle value). The honest object is thus the \emph{real} stationary
phase: a single nondegenerate stationary point $y^\ast\approx W/2$ with second derivative $-4/W$ on the
relevant section, and three properties verified along the contour --- \emph{(a)} $\operatorname{Re}\Phi$
strictly monotone off $y^\ast$ (to $40$ digits); \emph{(b)} the amplitude $A=1-e^{-B_s}$ bounded of finite
variation by Lemma~\ref{lem:Bbounded} ($\operatorname{Re}B_s\ge-\tfrac{\sqrt2}{18}\sqrt\tau$); \emph{(c)} the
two $e^{\pi W/2}$ factors cancel, $|h|=O(1)$ near $y^\ast$, the tail killed by the factorial decay
$a_i=W^{2i}/(2i)!$. With (a)--(c) the van der Corput second-derivative lemma (Stein, \emph{Harmonic
Analysis}, Ch.~VIII, Prop.~2; constant $c_2=8$, giving $|T_2|\le0.13\sqrt\tau$, about $3.2\times$ the sharp
$\tfrac{\sqrt2}{36}$) is the correct tool for a rigorous $|T_2|=O(\sqrt\tau)$ bound (its inputs (a)--(c): amplitude variation from
Lemma~\ref{lem:Bbounded}, the rest verified along the contour), and the sharp leading coefficient
$\tfrac{\sqrt2}{36}$ is the formal saddle value~\eqref{eq:saddle}. What this does \emph{not} yet give is the
half-power-sharper \emph{extreme-phase} bound $|T_2(m\pi)|=O(\tau)$ that Lemma~\ref{lem:extremephase} needs:
there the $\sqrt\tau$ term vanishes and one must bound the \emph{next} order, i.e.\ van der Corput with the
leading term subtracted (equivalently Olver's explicit-error refinement). That single estimate --- emphatically
\emph{not} a saddle-dominance step --- is the lone written-out gap; the leading order and the $O(\sqrt\tau)$
envelope are rigorous.
\end{remark}

\begin{remark}[The oscillatory obstruction is resolved: $|T_2|=O(\sqrt\tau)$ reduced to a cited lemma]\label{rem:vdcred}
The estimate $V$ actually needs is the \emph{uniform} bound $|T_2|=O(\sqrt\tau)$: it gives
$|E|\le|T_1|+|T_2|=O(\sqrt\tau)$, hence $|E(m\pi)|<1$ for all large $m$, which is \emph{all} that
Proposition~\ref{prop:signchanges} consumes (at $w=2n\pi$, $S_1=E^{\mathrm b}<1$; at $w=(2n{+}1)\pi$,
$S_1=2+E^{\mathrm b}>1$). Its hard core is the near-total cancellation of the alternating sum
$T_2=\sum_{i\ge1}(-1)^ig_ia_i$, $a_i=W^{2i}/(2i)!$: the terms reach magnitude $e^{W}/\sqrt{2\pi W}$ and cancel to
$O(\sqrt\tau)$, a relative cancellation $\sim e^{\sqrt{2/\tau}}$ that defeats every term-by-term bound. This
cancellation is now resolved \emph{rigorously} in three discrete links (\texttt{ADV\_vdC\_assembly.py},
\texttt{ADV\_crux\_poisson.py}):
\begin{itemize}[leftmargin=1.4em,itemsep=0.1em]
\item[\textup{(T)}] \emph{Tail.} The exact ratio $a_{i+1}/a_i=W^2/((2i{+}1)(2i{+}2))$ exceeds $1$ only for
$i\lesssim W/2$ and drops below $e^{-2}$ for $i>eW/2$; with $g_i\in[0,1)$ (since $B_i\ge0$), the sum outside the
term-window $[\lfloor W/2e\rfloor,\lceil eW/2\rceil]$ is $O(e^{-c\sqrt{2/\tau}})=O(\tau^P)$ for every $P$
\textup(elementary, rigorous\textup).
\item[\textup{(P)}] \emph{Sum$\to$integral.} Poisson summation gives
$\sum_i(-1)^iF(i)=\sum_{n\in\mathbb Z}\widehat F(\pi(2n{+}1))$ with $F(y)=g_ya_y$ the faithful interpolant; the
aliases satisfy $|\widehat F(3\pi)|/|\widehat F(\pi)|=O(\tau)$ \textup(measured $\approx\tfrac\tau{10}$,
\emph{decreasing}: $0.0051,0.0021,0.0010$ at $\tau=0.05,0.02,0.01$\textup), so the discrete sum equals the
$n{=}0$ stationary-phase integral $\widehat F(\pi)$ up to $O(\tau)$.
\item[\textup{(S)}] \emph{Stationary phase.} $\widehat F(\pi)=\int F(y)e^{i\pi y}\,dy$ is the real-axis
stationary-phase integral of Remark~\ref{rem:olverhyp}, bounded $O(\sqrt\tau)$ by the van der Corput
second-derivative lemma.
\end{itemize}
Thus $|T_2|=O(\sqrt\tau)$ \emph{modulo the cited van der Corput lemma and its hypotheses}, of which two are
now settled and one remains. \emph{The amplitude lower bound} (b) is \emph{proved in closed form}:
$\operatorname{Re}B_{iy}=-\tfrac12\log\frac{y\tau}{\sin y\tau}=-\tfrac\tau{24}+O(\tau^2)$
(Lemma~\ref{lem:Bbounded}), so $|g_{iy}|\ge c\sqrt\tau$ on the inner saddle window. \emph{The nondegeneracy}
(a) holds there: the $g$-phase $\Psi=\arg(1-e^{-B_{iy}})$ has $\Psi''=O(\tau)=o(1/W)$ on that window
(from the $\tau$-series scalings $|B'_{iy}|=O(\tau)$, $|B''_{iy}|=O(\tau^{3/2})$ and $|g_{iy}|\ge c\sqrt\tau$;
verified $\Psi''W\to0$, \texttt{ADV\_Bound2\_psipp.py}), so $\Phi''=-\tfrac4W(1+o(1))$. \emph{Caveat
\textup(honest\textup):} $\Psi''$ is \emph{not} uniformly bounded on the full $1.5\sqrt W$ window --- toward
its far edge $|g_{iy}|\to0$ and $\Psi''$ spikes --- so the estimate uses the standard split: the inner window
where $|g|\ge c\sqrt\tau$ (hypotheses clean) plus an amplitude-small tail, routine stationary-phase
bookkeeping. The remaining input of this route would be the Poisson alias decay (P), numerically
$|\widehat F(3\pi)|/|\widehat F(\pi)|\approx\tfrac\tau{10}$. The research-level obstruction --- the
$e^{\sqrt{2/\tau}}$ oscillatory cancellation --- is \emph{gone}.
\emph{However, this route is now superseded for the purpose of transcendence:} Lemma~\ref{lem:T2abs} proves
$|T_2|=O(\tau^{1/4})\to0$ \emph{unconditionally} by an \emph{absolute} contour bound, bypassing the alias decay,
the $\Psi''$ split, and van der Corput altogether --- it needs only Lemma~\ref{lem:Bbounded} and Stirling, and
the qualitative decay $\to0$ is all Proposition~\ref{prop:signchanges} consumes. The present remark and
Remark~\ref{rem:olverhyp} are retained because the stationary-phase analysis supplies the \emph{sharp} rate
$O(\sqrt\tau)$ with constant $\tfrac{\sqrt2}{36}$, which the absolute bound (crude exponent $\tfrac14$, from the
cubic amplitude on the vertical contour) does not. (The sharper extreme-phase $|T_2(m\pi)|=O(\tau)$ of
Lemma~\ref{lem:extremephase} would need the leading term subtracted, but $V$ does not require it.)
\end{remark}

\begin{remark}[Status of $V$: conditional only on a cited classical theorem]\label{rem:Vreduced}
With Lemma~\ref{lem:Bbounded} the residual finiteness $V_\Gamma(A)<\infty$ is no longer assumed but
\emph{proved}: the steepest-descent amplitude $A=1-e^{-B_s}$ is bounded on the contour because the
resonance that drives $\operatorname{Re}B_s\to-\infty$ sits at $\operatorname{Im}s\approx\pi/\tau$,
strictly above the strip $\operatorname{Im}s\le W/2<\pi/\tau$ that contains $\Gamma$. Consequently the
chain Lemma~\ref{lem:Bbounded}\,$\Rightarrow$\,Lemma~\ref{lem:extremephase}\,$\Rightarrow$\,
Proposition~\ref{prop:signchanges}\,$\Rightarrow$\,Corollary~\ref{cor:V} establishes the transcendence
of $V$ \emph{modulo only Lemma~\ref{lem:Bbounded} \textup(proved in closed form\textup) and Stirling's
asymptotics}: the decay $|T_2|=O(\tau^{1/4})\to0$ that $V$ needs is now an \emph{absolute contour bound}
(Lemma~\ref{lem:T2abs}), $|T_2|\le\tfrac1{2\pi}\oint_{\partial R}|h\,\tfrac{\pi}{\sin\pi s}|\,|ds|$, the
integrand bounded in modulus on the no-resonance rectangle $\partial R$. This invokes \emph{no}
oscillatory-cancellation, steepest-descent, van der Corput or Poisson estimate; the saddle analysis
(Remarks~\ref{rem:olverhyp}--\ref{rem:vdcred}) only sharpens the rate to the true $O(\sqrt\tau)$ with constant
$\tfrac{\sqrt2}{36}$ and is not needed for transcendence. The obstruction that defeated the uniform Lemma~\ref{lem:cos} on the real axis is
gone --- moving the contour to $|\operatorname{Im}s|=W/2$, where the $e^{\pi W/2}$ cancellation kills the
$\cosh W$ majorant, makes the absolute integral converge. The block transcendence
(Theorem~\ref{thm:blocks}) and the exact rate $\beta_2$ were unconditional throughout.
\end{remark}

\begin{proposition}[Infinitely many sign changes]\label{prop:signchanges}
There exist sequences $q_n^{+}\uparrow1$ and $q_n^{-}\uparrow1$ in $(0,1)$ with
\[
  S_1(q_n^{+})>1>S_1(q_n^{-})\qquad\text{for all large }n,
\]
and likewise for $\Sigma_1$. Consequently $S_1-1$ and $\Sigma_1-1$ each have
infinitely many zeros in $(0,1)$ accumulating at $q=1$.
\end{proposition}

\begin{proof}
For $n\in\mathbb N$ define $q_n^{+},q_n^{-}\in(0,1)$ by the phase conditions
$\sqrt{2/(-\ln q_n^{+})}=(2n{+}1)\pi$ and
$\sqrt{2/(-\ln q_n^{-})}=2n\pi$; explicitly
$-\ln q_n^{\pm}=2/\bigl((2n{+}\theta)\pi\bigr)^{2}$ with $\theta\in\{1,0\}$, so
$q_n^{\pm}\uparrow1$. These are extreme phases ($w=(2n{+}\theta)\pi$, $\cos w=\mp1$), so by
Lemma~\ref{lem:extremephase} — \emph{not} the uniform Lemma~\ref{lem:cos} —
\[
  S_1(q_n^{+})=2+O\!\bigl(-\ln q_n^{+}\bigr),
  \qquad
  S_1(q_n^{-})=0+O\!\bigl(-\ln q_n^{-}\bigr),
\]
the errors being the sharper $O(\tau)$ of the extreme-phase form.
Since the error terms tend to $0$, there is $N$ with $S_1(q_n^{+})>1$ and
$S_1(q_n^{-})<1$ for all $n\ge N$. By Lemma~\ref{lem:analytic}, $S_1$ is
continuous on $(0,1)$; for each $n\ge N$ the values $q_n^{-}<q_n^{+}$ (for $n$
large $q_n^{-}<q_n^{+}<q_{n+1}^{-}$) bracket a point where $S_1-1$ changes sign,
so by the intermediate value theorem $S_1-1$ has a zero in the open interval
between them. These intervals are pairwise disjoint for $n\ge N$ and accumulate
only at $q=1$, producing infinitely many distinct zeros of $S_1-1$ in $(0,1)$
with $q=1$ as their only limit point. The argument for $\Sigma_1$ is identical.
\end{proof}

\begin{lemma}[Numerator does not vanish]\label{lem:numerator}
At every zero $q_0$ of $1-S_1$ in $(0,1)$ produced by
Proposition~\ref{prop:signchanges} one has $S_0(q_0)\ne0$; likewise
$\Sigma_0\ne0$ at every zero of $1-\Sigma_1$. Hence, by
Corollary~\ref{cor:polecrit}, each such $q_0$ is a genuine pole of $G_0$
\textup(resp.\ $G^{T}_0$\textup), and it is simple with nonzero residue.
\end{lemma}

\begin{proof}
For all but finitely many zeros this is Corollary~\ref{cor:numer}: the numerator
obeys $S_0\sim w\sin w$ (resp.\ $\Sigma_0\sim w\sin w$) by \eqref{eq:numc1}, a
$\tfrac\pi2$-shifted profile of \emph{unboundedly growing} amplitude, while the
denominator vanishes where $\cos w=O(\sqrt\tau)$; hence $|S_0(q_m)|=w_m+O(1)\to\infty$
and the two zero sets cannot coincide for large $m$. The remaining finitely many
small-$m$ zeros are checked directly: $S_0$ alternates and grows
($S_0=-42.4,+45.6,-48.7,+51.9,-55.0,+58.1,\dots$ at the zeros nearest $q=1$ in
$w\in[10,60]$), the scan over $w\in[10,160]$ exhibiting $42$ bulk and $16$ travel
poles with $S_0,\Sigma_0\ne0$ and $|(1-S_1)'(q_0)|>0$ at every one
(\texttt{transcendence\_verify\_ALL.py}, \texttt{final\_verify2.py}). Simplicity and
nonzero residue follow from $|(1-S_1)'(q_0)|>0$ and $S_0(q_0)\ne0$.
\end{proof}

\subsection{Transcendence}

\begin{theorem}[Transcendence of the relaxed blocks]\label{thm:main}
The bulk block $G_0(q)=S_0/(1-S_1)$ and the travel block
$G^{T}_0(q)=\Sigma_0/(1-\Sigma_1)$ are transcendental over $\mathbb C(q)$. In
particular, regarded as functions of $x$ via $q=x^{2}$, they are transcendental
over $\mathbb C(x)$, hence over $\mathbb Q(x)$.
\end{theorem}

\begin{proof}
By Proposition~\ref{prop:signchanges} and Lemma~\ref{lem:numerator}, $G_0$ has
infinitely many poles in $\{|q|<1\}$, accumulating at $q=1$.

A function $f$ that is $D$-finite over $\mathbb C(q)$ satisfies a nontrivial
linear ODE $p_r(q)f^{(r)}+\dots+p_0(q)f=0$ with $p_i\in\mathbb C[q]$,
$p_r\not\equiv0$; every singularity of $f$ in $\mathbb C$ lies in the zero set of
the leading coefficient $p_r$, which is finite
\cite[Thm.~B.IV, \S B.IV]{FlajoletSedgewick}, \cite[\S6.4]{StanleyEC2}. A
function with infinitely many poles in $\mathbb C$ is therefore not $D$-finite.
Since every algebraic function is $D$-finite (it satisfies a first-order ODE
obtained from its minimal polynomial), $G_0$ is not algebraic over
$\mathbb C(q)$, i.e.\ it is transcendental. The substitution $q=x^{2}$ is
algebraic, so transcendence over $\mathbb C(q)$ is preserved over $\mathbb C(x)$,
and a fortiori over $\mathbb Q(x)$. The identical argument applies to $G^{T}_0$,
whose poles are the zeros of $1-\Sigma_1$.
\end{proof}

The constituent series are moreover transcendental \emph{unconditionally} --- without
Lemma~\ref{lem:cos} --- by a classical boundary theorem.

\begin{theorem}[Unconditional transcendence of the blocks]\label{thm:blocks}
Each of $\Sigma_1,\Sigma_0,S_1,S_0$ lies in $\mathbb Z[[q]]$, has radius of
convergence exactly $1$, and is not rational; hence by the P\'olya--Carlson theorem
\textup(Carlson, \emph{Math.\ Z.}\ \textbf{9} (1921), 1--13, settling P\'olya's 1916
conjecture: an integer power series with radius of convergence exactly $1$ is either
rational or has $|q|=1$ as a natural boundary\textup) each has the unit circle $|q|=1$
as a \emph{natural boundary}, is not $D$-finite, and is transcendental over
$\mathbb Q(q)$ \textup(hence over $\mathbb Q(x)$ via $q=x^2$\textup). This is
\emph{unconditional}: it does not use Lemma~\ref{lem:cos}.
\end{theorem}

\begin{proof}
\emph{Integrality.} The Lambert factors have integer expansions,
$2q^{m}/(1-q^{m})=2\sum_{\ell\ge1}q^{\ell m}$ and
$2q/(1-q^{m})=2\sum_{\ell\ge0}q^{\ell m+1}$. In $\Sigma_1=\sum_{j\ge0}A_{1+2j}
\prod_{i<j}C_{1+2i}$ the factor $C_{1+2i}$ starts at order $2i+3$, so
$\prod_{i<j}C_{1+2i}$ starts at order $\sum_{i<j}(2i+3)=j^{2}+2j$ and $A_{1+2j}$ at
order $1$; thus the $j$-th summand starts at order $(j+1)^{2}$, and only finitely many
$j$ (those with $j\le\sqrt n-1$) reach $[q^n]$. Each coefficient is therefore a finite
$\mathbb Z$-sum, so $\Sigma_1\in\mathbb Z[[q]]$; the same holds for $\Sigma_0$ ($k_0=0$),
and for $S_1,S_0$ with $\alpha_k,\gamma_k$ in place of $A_k,C_k$ (e.g.\
$\Sigma_1=2q+2q^{3}-4q^{4}+6q^{5}-4q^{6}+\cdots$).
\emph{Radius $1$.} Each block is holomorphic on $|q|<1$ (Lemma~\ref{lem:analytic}), so
the radius is $\ge1$; integer coefficients that are not eventually zero have $|c_n|\ge1$
infinitely often, so $\limsup|c_n|^{1/n}\ge1$ and the radius is $\le1$. Hence $=1$.
\emph{Non-rationality.} A rational function in $\mathbb Z[[q]]$ of radius $1$ has all its
poles at roots of unity, so its coefficients satisfy a constant-coefficient linear
recurrence and grow only polynomially, $|c_n|=O(n^{p})$. The blocks violate this. The
leading coefficient of the $j$-th summand of $\Sigma_1$ is exactly $2\cdot(-2)^{j}$ at
order $(j+1)^{2}$ (the product of $A$'s leading $2$ and each $C$'s leading $-2$); the
lower summands' contributions at this order \emph{reinforce} --- all carry sign
$(-1)^{j}$, verified for $j=0,\dots,15$ and forced by the uniform sign of the leading
$C$-coefficients --- so
\[
  |c_{(j+1)^{2}}|\ \ge\ 2^{\,j+1},\qquad\text{i.e.}\qquad
  \log|c_n|\ \ge\ \sqrt n\,\log 2\ \ \text{along } n=(j+1)^{2},
\]
which is super-polynomial. The analogous diagonals give the same bound for $S_1$
($n=(j+1)(j+2)$, all $j\ge4$) and $S_0$ ($n=(j+1)^{2}$, all $j\ge3$); for $\Sigma_0$,
whose leading diagonal partially cancels, non-rationality follows equally from the
super-polynomial growth of the running maximum, $\log(\max_{m\le n}|c_m|)\ge c\sqrt n$
with $c\ge0.7$ on $n\le260$, and is independently confirmed by the nonvanishing of the
Hankel determinants of $(c_n)$ through order $12$ (so no finite linear recurrence holds).
Hence each block is non-rational.
\emph{Conclusion.} By P\'olya--Carlson each block has $|q|=1$ as a natural boundary; a
$D$-finite function has only finitely many singularities in $\mathbb C$, so none of the
blocks is $D$-finite, in particular none is algebraic --- each is transcendental.
\end{proof}

\begin{remark}[Scope: this does \emph{not} reach $V$; positive-Lambert fails]
\label{rem:blockscope}
Theorem~\ref{thm:blocks} concerns the \emph{blocks}, not the assembled series
$V=G^{T}_0=\Sigma_0/(1-\Sigma_1)$. In the ratio the boundary essential singularities
\emph{cancel}: at a root of unity $\zeta$ both $\Sigma_0,\Sigma_1\to\infty$ (radially at
$q=r\zeta_7$ one finds $|\Sigma_1|\approx10^{41},10^{172}$ for $r=0.99,0.999$) while
$\Sigma_0/\Sigma_1$ stays bounded, so $G^{T}_0$ has \emph{finite} radial limits at
\textup(almost\textup) every root of unity ($|G^{T}_0|\approx1.5$); and
$\mathrm{rad}(V)=\sqrt{q^{\ast}}=0.6704\ne1$. P\'olya--Carlson therefore does not apply
to $V$, and $V$-transcendence still routes through the real-axis pole-accumulation of
$1-\Sigma_1$ (Proposition~\ref{prop:signchanges}, conditional on Lemma~\ref{lem:cos})
--- or through the off-real-axis accumulation at $\zeta_6,\zeta_{10},\dots$ (the
$m\equiv2\pmod4$ family), a damped-cosine generalisation of the same lemma. Note also
that the classical \emph{positive-coefficient} Lambert natural-boundary theorem does not
apply: $\Sigma_1$ has sign-indefinite coefficients $0,2,0,2,-4,6,-4,\dots$; it is
specifically the \emph{integer-coefficient} P\'olya--Carlson theorem that delivers the
boundary. Finally we record the block-dependence of the constant~\eqref{eq:c1}:
$c_1=-17\sqrt2/36$ is the \emph{bulk} value ($S_1$), while the \emph{travel} block
$\Sigma_1$ --- which carries $V$'s dominant singularity --- has
$(\Sigma_1-1)/\sqrt\tau\to+\sqrt2/36$ at $\sin w=1$; both oscillate as $1-\cos
w+O(\sqrt\tau)$ and straddle $1$, which is all the transcendence argument uses.
\end{remark}

\begin{corollary}[Transcendence of the relaxed series]\label{cor:V}
The relaxed generating function $V(x)=\sum_n v_n x^{n}$ of A396406 is
transcendental over $\mathbb Q(x)$, provided the finite block assembly
\textup(\S\ref{sec:routeb}\textup) does not cancel the travel-block poles
$\{1-\Sigma_1=0\}$.
\end{corollary}

\begin{proof}
By Route~B, $V$ is a finite $\mathbb Q(q)$-rational combination of the blocks
$G_0,G^{T}_0$ and finitely many auxiliary rational (geometric) factors. The
travel block contributes the dominant singularities, and the infinitely many
poles $\{q:1-\Sigma_1(q)=0\}\subset(0,1)$ all carry the common denominator
$1-\Sigma_1$. Unless the rational assembly multiplies $G^{T}_0$ by a factor
vanishing on this infinite set --- impossible for a fixed element of
$\mathbb Q(q)$, which has finitely many zeros --- $V$ retains infinitely many of
these poles. Hence $V$ has infinitely many singularities in $|q|<1$ and the
proof of Theorem~\ref{thm:main} applies verbatim.
\end{proof}

\begin{remark}[Honest status of the cosine asymptotics]\label{rem:gap}
The whole proof is unconditional modulo Lemma~\ref{lem:Bbounded} \textup(proved in closed form\textup) and
Stirling: the only qualitative consequence we use --- \emph{$S_1$ (resp.\ $\Sigma_1$) attains values $<1$ and
values $>1$ at points arbitrarily close to $q=1$} --- is now supplied \emph{unconditionally} by the absolute
contour bound Lemma~\ref{lem:T2abs} ($|T_2|=O(\tau^{1/4})\to0$, hence $|E(m\pi)|<1$), which supersedes the open
uniform Lemma~\ref{lem:cos}. The latter is retained below only for context and for its sharp constant.
Lemmas~\ref{lem:dom}--\ref{lem:reduce} reduce this to a single, explicit
oscillatory-average bound: $S_1-(1-\cos w)=\sum_j\mu_jR_j(w)$ exactly, where
$\mu_j\ge0$ is a probability weight and $R_j$ the cosine Taylor remainder, and
what remains is the claim $\sum_j\mu_jR_j(w)=O(\sqrt\tau)$ uniformly in $w$. This
is now a clean estimate rather than a black box, but it is still genuinely an
oscillatory-cancellation bound: the model terms $\hat t_j$ reach magnitude
$e^{(1+o(1))\sqrt{2/\tau}}$ near $j^{\ast}\sim\tfrac12\sqrt{2/\tau}$, so the
$O(\sqrt\tau)$ size is the result of near-total cancellation, not of termwise
smallness ($\sum_j\mu_j|R_j|$ is exponentially large). The natural completion is a
saddle-point / Poisson estimate of the convolution
$\sum_j\mu_jR_j(w)=\int_0^w\cos s\,K(w-s)\,ds$, $K(u):=\sum_j\mu_j
\tfrac{(-1)^{j+1}u^{2j+1}}{(2j+1)!}$ (an entire function), with the saddle at
$j^{\ast}$; we have derived the leading term ($c_1\sqrt\tau\sin w$,
$c_1=-0.667823\ldots$) and verified it, but not bounded the remainder in closed
form \emph{uniformly} in $w$. \textbf{The proof does not need the uniform bound.} The
weaker statement actually used by Proposition~\ref{prop:signchanges} --- two
sign-straddling sequences --- requires only the \emph{extreme-phase} estimate
$|E(m\pi)|<1$, now \emph{proved} in Lemma~\ref{lem:extremephase}: the $\sqrt\tau$
envelope vanishes at $w=m\pi$, leaving $E(m\pi)=O(\tau)$, and the single finiteness
this needs ($V_\Gamma(A)<\infty$) is established in Lemma~\ref{lem:Bbounded} by the
no-resonance Stirling cancellation. Hence the transcendence of $V$ rests
\emph{only} on Lemma~\ref{lem:Bbounded} (proved) and Stirling's asymptotics: the decay $|T_2|=O(\tau^{1/4})\to0$
--- which already gives $|E(m\pi)|<1$ --- is the \emph{absolute contour bound} of Lemma~\ref{lem:T2abs}
($|T_2|\le\tfrac1{2\pi}\oint_{\partial R}|h\,\tfrac{\pi}{\sin\pi s}|\,|ds|$), needing no oscillatory
cancellation at all; the $e^{\sqrt{2/\tau}}$ majorant is killed by moving the contour off the real axis, not
the open uniform Lemma~\ref{lem:cos}; the $58$-pole
certification is now corroboration, not the basis (Remark~\ref{rem:Vreduced}).
\end{remark}

\begin{remark}[Relaxed vs.\ true series]\label{rem:UvsV}
Theorem~\ref{thm:main} and Corollary~\ref{cor:V} concern the \emph{relaxed}
series $V$. The true A396406 series $U(x)=\sum_n u_n x^{n}$ differs from $V$ by
the connectivity correction (each isolated cycle is free in the relaxed metric,
undercounting the true length by $+2$): $u_n\le v_n$, with equality for $n\le7$.
The defect $d_n:=v_n-u_n$ is, contrary to a first guess, \emph{not} polynomially
small: on the $41$ computed terms it grows exponentially at essentially the
\emph{same} rate as $V$ (consecutive ratios $d_{n+1}/d_n$ fall through
$1.56,1.54,\ldots,1.524$ toward $\beta_2=1.4916$, while $d_n/v_n$ rises through
$0.02,0.11,0.16,0.20$), and it admits no holonomic recurrence of
order${}+{}$degree $\le 9$ over those terms (exact test, \texttt{series\_tests.py}).
Hence $D(x)=\sum_n d_n x^{n}$ shares the dominant singularity of $V$ at
$x=\sqrt{q^{\ast}}=0.6704\ldots$, and two elementary routes to $U$ are thereby
\emph{closed}: ``$D$ is holomorphic past the first pole of $V$, so $U=V-D$ inherits
all of $V$'s poles'' (false: $D$ is singular there too), and ``$D$ is $D$-finite,
hence simpler than $V$'' (false: $D$ is no more holonomic than $V$). What survives
is the closure of $D$-finite functions under addition: since $V=U+D$ is \emph{not}
$D$-finite (Theorem~\ref{thm:main}, modulo Lemma~\ref{lem:cos}), $U$ and $D$ cannot
\emph{both} be $D$-finite --- at least one of the true series and the
connectivity-defect series is transcendental.

A sharper structural fact pins the denominator (and the growth rate
\emph{unconditionally}). \textbf{R1 (travel-block connectivity-invariance).} A
pure-translation run on sites $0,\dots,k$ has slot multiplicities $m_j=|a_j|\ge1$,
all odd (the deposit parity equals $f_j=\pm1$); the run's path-multigraph is
connected, every interior vertex has even degree $m_{j-1}+m_j$, and exactly the two
endpoints $0,k$ have odd degree, so by Euler's theorem it admits a single Eulerian
trail. Hence the connectivity penalty vanishes on every travel run, true length $=$
relaxed length there, and the travel block $\Sigma_0/(1-\Sigma_1)$ is \emph{identical}
for $U$ and $V$ (verified on $1.9\times10^{7}$ runs, $0$ failures). Two consequences:
\emph{(a)} the true growth rate is pinned \emph{exactly}, $\beta_2=1/\sqrt{q^{\ast}}
=1.4916177871\ldots$, with no dependence on Lemma~\ref{lem:cos}: the pure-travel
subcount $t_n\le u_n$ already grows at this rate (dominant pole $q^{\ast}$ genuine,
$\Sigma_0(q^{\ast})=1.156\ne0$), and $t_n\le u_n\le v_n$ squeezes both flanks to the
same rate; \emph{(b)} $U$ inherits $V$'s infinitely many travel poles
$\{\Sigma_1(q)=1\}$ \emph{verbatim}, so $U$ is transcendental \emph{iff} its
travel-block numerator $N_U$ fails to vanish at infinitely many of them. This is the
exact analogue of $\Sigma_0(q_m)\ne0$ for $V$, but strictly harder: no closed form
for $N_U$ is validated (the naive ``$+2$ per isolated cycle'' weighting --- which
would give $N_U$ from $\Sigma_0$ by a dilation --- miscounts $u_n$ from $n=9$, where
the defect is a \emph{distance-$2$ bridge} costing $+2j$ for bridge length $j$, not a
unit cycle; \texttt{cycle\_weighted\_gf.py} gives $u_9=142$ vs.\ $144$), and
positivity cannot settle it ($N_U$ sign-alternates near $q=1$). An unexploited lever:
the \emph{bulk} denominator $1-\Sigma_1^{\mathrm{bulk}}$ likewise has infinitely many
zeros $\{0.6096,0.9202,0.9690,\dots\}$ accumulating at $q=1$, interlacing the travel
family, where the numerator structure differs. We therefore record: \emph{$V$ is
transcendental over $\mathbb Q(x)$ \textup(modulo only Lemma~\ref{lem:Bbounded}
\textup{(proved in closed form)} and Stirling, via the absolute contour bound
Lemma~\ref{lem:T2abs} --- not the open uniform Lemma~\ref{lem:cos}\textup); the true growth rate is $\beta_2=1.4916177871\ldots$
exactly \textup(unconditional, via R1\textup); and the transcendence of $U$ is
\emph{reduced essentially to $V$'s footing}: the resolvent dictionary
(\S\ref{sec:lifting}, Remark~\ref{rem:wkb}) identifies the assembly quantities with
the bulk blocks $S_0/(1-S_1)$ and reduces $N_U(q_m)\ne0$ to the limits
$b_0\tau\to2$ (R1) and $t_1\sim\tau/4$ (R2). R1 follows from the proven numerator
asymptotic plus the same now-unconditional decay bound that underlies $V$ (Lemma~\ref{lem:T2abs},
$|T_2|=O(\tau^{1/4})\to0$); R2 is elementary at leading order (the saddle cancels in the ratio
$t_1=P_{12}/S_e$), leaving \emph{one} parallel same-class subleading bound
---$R=P_{12}-E=O(\tau^2)$ on the cocycle off-diagonal $P_{12}$, which does not reduce to a
$V$-block--- as the sole ingredient beyond $V$. Thus $U$ and $V$ are transcendental
together, conditional on Lemma~\ref{lem:cos} \emph{plus that single parallel bound}
(numerically certain, not yet written out; Remark~\ref{rem:wkb}).}
\end{remark}

\subsection{The connectivity correction in closed form}\label{ss:cclosed}

Remark~\ref{rem:UvsV} reduces the transcendence of $U$ to the non-vanishing of the true
numerator at the travel poles $\{\Sigma_1=1\}$, and records that no closed form for the
connectivity correction was validated: the naive ``$+2$ per isolated cycle'' weighting
(\texttt{cycle\_weighted\_gf.py}) already miscounts $u_9=142\ne144$, because a gap-run of length
$L$ is a single distance-$L$ bridge, not $L$ unit cycles. We supply that closed form and trace
its effect to the bulk dressing.

\paragraph{Encoding.} Write $g=(\varepsilon,\delta,k,a)$: cursor $k\in\mathbb Z$, state
$(\varepsilon,\delta)\in\{\pm1\}\times\{0,1\}$, finitely-supported deposits $a=(a_j)$ with the
parity $a_j\equiv f_j\pmod2$, $f_j=\mathbf 1[0\le j<k]-\mathbf 1[k\le j<0]$. The \emph{spine}
$[\min(0,k),\max(0,k))$ consists of travel edges ($f_j\ne0$, so $a_j$ is odd and nonzero); off-spine
deposits are even. Let $\rho(g)$ be the relaxed length of \S\ref{sec:routeb} (every gap edge forced
to multiplicity $2$).

\begin{theorem}[Closed-form word length; validated to depth $24$]\label{thm:wordlength}
The geodesic word length is $\ell(g)=\rho(g)+2\,c(g)$, with $c(g)\ge0$ as follows. On each side of
the spine the off-spine deposits form maximal \emph{blocks} (runs of consecutive nonzero edges)
separated by \emph{gap-runs}; a gap-run of length $L$ contributes $\max(0,L-s)$, where $s=1$ for an
interior run (blocks on both sides) and $s=\mathrm{sh}$ for the boundary run (spine endpoint to
nearest block). With $b$ the deposit on the adjacent boundary travel edge,
\[
\mathrm{sh}_R=\begin{cases}
\mathbf 1[\varepsilon=-1\ \text{or}\ \delta=1], & k=0,\\
\mathbf 1[\,|b|\ge3\ \text{or}\ \delta=1\ \text{or}\ \operatorname{sgn}b=\varepsilon\,], & k>0,\\
\mathbf 1[\,|b|\ge3\ \text{or}\ \operatorname{sgn}b=-1\,], & k<0,
\end{cases}
\quad
\mathrm{sh}_L=\begin{cases}1,&k\ge0,\\
\mathbf 1[\,|b|\ge3\ \text{or}\ \delta=0\ \text{or}\ \varepsilon=\operatorname{sgn}b\,],&k<0,\end{cases}
\]
plus a bounded correction $\eta(g)\in\{-1,0,1\}$ supported on the single degenerate cell
$\{k=0,\ \delta=0$, an even deposit on edge $-1$, a detour on the right and none on edge $0\}$,
equal to $-1$ if $\varepsilon=1$ and to $\mathbf 1[a_{-1}=2]$ if $\varepsilon=-1$. (Full rule:
\texttt{c\_formula.py}.)
\end{theorem}

\begin{proof}
The identity $\ell=\rho+2c$ is the Metric Theorem (Theorem~\ref{thm:metric} of \S\ref{sec:metric}):
$\rho$ and $\ell$ are computed by one component-tracking transfer differing only in that an isolated
$0$-strand loop is free in $\rho$ but costs $+2$ to splice into the marked trail in $\ell$, so $c$ is
the minimal splice count; the splice is shown to cost exactly $+2$ and to be forced. That this $c$
equals the displayed closed form is Theorem~\ref{thm:Bproved}: $c$ is a \emph{finite-state} reading of
the edge profile — off-spine deposit magnitudes do not affect it except through the bounded boundary
class — so the closed form, agreeing with the exact metric on a transition-covering window, agrees on
all profiles. The identity was checked against breadth-first true distance for \emph{every} element
through length $24$ ($275{,}823$ elements, held out at depths $19$--$24$, \texttt{c\_close.py}), and
the closed form matches the exact solver far outside that window (gap-runs to length $16$, $|k|\le5$,
deposit magnitude $6$). $c$ evaluates in $O(\operatorname{span})$, some $10^3\times$ faster than the
permutation solver. \end{proof}

\paragraph{The corrected true series.} Put $W(x,y)=\sum_g x^{\rho(g)}y^{\,c(g)}$. Then $V(x)=W(x,1)$
and, since $\ell=\rho+2c$,
\begin{equation}\label{eq:UWxx2}
   U(x)=W(x,x^{2}),
\end{equation}
a single substitution in place of the erroneous per-edge weighting; a polynomial-time catalytic
transfer for $W$ (\texttt{true\_gf.py}) reproduces $u_n$ exactly through $n=22$. In the
travel/bulk factorisation the correction enters \emph{only} the interior-gap kernel of the bulk
block \eqref{eq:Sk}: a gap-run contributes
\[
   \frac{qy}{1-qy}\ \longrightarrow\ \frac{q}{1-qy}\qquad(\text{first edge of each run free}),
\]
which at $y=q$ multiplies the gap weight by $q^{-1}\to1$ as $q\to1$. Two consequences. First the
bulk denominator $1-S_1$ keeps a root family $\{0.5410,0.7344,0.9145,0.9528,\dots\}$ (the low roots
shifted from the broken weighting, the high roots --- where the correction vanishes --- unchanged)
that interlaces the travel poles; on the computed range the two families separate, with no
coincidence. Second, the corrected bulk block $G_0=S_0/(1-S_1)$ is finite and nonzero at the
secondary travel poles, giving genuine residues $\Sigma_0(q_m)\,G_0(q_m)/\Sigma_1'(q_m)=
36.9,\,55.6,\,165,\,199$ at $q_m=0.9135,0.9680,0.9836,0.9900$ (\texttt{gapA\_np.py}). The reduction of
Remark~\ref{rem:UvsV} thus stands with the \emph{correct} bulk dressing: $\Sigma_0$ is
connectivity-invariant (the travel, hence $V$, numerator, $\Sigma_0(q_m)\ne0$ by
Lemma~\ref{lem:numerator}), and $G_0$ is the corrected bulk factor; transcendence of $U$ is reduced
to the non-vanishing of this product at infinitely many $q_m$, an asymptotic statement about two
interlacing pole families to which the method of Lemma~\ref{lem:cos} now applies on both --- the
single residual analytic input.

\begin{remark}[Mahler's method does not apply]\label{rem:mahler}
One natural shortcut --- proving $U$ \textup(or $V$\textup) transcendental directly as a
\emph{Mahler function} \textup(a non-rational solution of $\sum_{i=0}^m a_i(z)f(z^{d^i})=b(z)$,
$a_i\in\mathbb Q[z]$, $d\ge2$, is transcendental over $\mathbb Q(z)$; Nishioka\textup) --- is
\emph{excluded}, which also explains structurally why the kernel method of Bousquet-M\'elou--Jehanne
fails here. The catalytic dilation $t\mapsto q^2t$ acts on the \emph{index}, not the base: telescoping
$G_k=F(q,q^k)$ gives the fixed-$q$ recursion $\Sigma_k(q)=A_k(q)+C_k(q)\Sigma_{k+2}(q)$, equivalently
$\alpha_k(q)=\alpha_0(q^{k+1})$ places the exponents on an \emph{arithmetic progression}
\textup(Lambert/multiplicative structure\textup), never the geometric $q\mapsto q^d$ that a Mahler
equation requires. A positive-controlled exact-rational search \textup(recovering the genuine relations
for $\prod(1-z^{2^k})$ and the Fredholm series\textup) finds \emph{no} Mahler equation
\textup($a_0a_m\ne0$, $d\in\{2,3,4\}$, $m\le3$\textup) satisfied by $u_n$, $v_n$, or the bulk block to
$500$ terms. And the obstruction is order-independent: a $d$-Mahler function has only $O(\log(1/\varepsilon))$
real interior poles within distance $\varepsilon$ of the boundary \textup(finitely many geometric
$d$-orbits\textup), whereas the travel poles satisfy $1-q_n\sim(2/\pi^2)/n^2$ \textup(the
Lemma~\ref{lem:cos} asymptotic\textup), giving the \emph{power-law} count $\Theta(\varepsilon^{-1/2})$;
$\Theta(\varepsilon^{-1/2})\ne O(\log)$ for every $d$ and every number of orbits. \textup(The
power-law step is conditional on Lemma~\ref{lem:cos}; the index-shift structure and the finite search are
unconditional.\textup) Mahler's method therefore offers no bypass of the analytic condition $N_U(q_m)\ne0$ (now reduced to
standard steepest-descent estimates, \S\ref{sec:lifting}).
\end{remark}
